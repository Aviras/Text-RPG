38      public boolean activate(Wezen actor, Wezen target, int[] actorPos, int[] targetPos, String aimFor, HashMap<double[][], Double> covers, double[] coefficients){
39          
40          logger.info("Activating ability " + name + " by " + actor.getName());
41          //TODO check requirements
42          //checks for cooldown
43          if(actor.getAbilities().get(this) == 0){
44              //distance is now independent of window size, only relative size. Width of map is always defined as 15m.
45              double meter = GameFrameCanvas.dungeonMap.getWidth()/15;
46              double distance = new Point(actorPos[0],actorPos[1]).distance(targetPos[0],targetPos[1])/meter;
47              if(distance >= range[0]){
48                  if(distance <= range[1]){
49                      try {
50                          
51                          Global.beanShell.set("actor", actor);
52                          Global.beanShell.set("target", target);
53                          Global.beanShell.set("actorPos", actorPos);
54                          Global.beanShell.set("targetPos", targetPos);
55                          Global.beanShell.set("actionTarget", aimFor);
56                          Global.beanShell.set("enemyPortrait", GameFrameCanvas.enemyPortrait);
57                          Global.beanShell.set("dungeonMap", GameFrameCanvas.dungeonMap);
58                          Global.beanShell.set("distance", distance);
59                          Global.beanShell.set("covers", covers);
60                          Global.beanShell.set("hitAbility", Data.abilities.get(0));
61                          Global.beanShell.set("RPGMain",new RPGMain());
62                          Global.beanShell.set("hostA", new HostileArea());
63                          Global.beanShell.set("Global", new Global());
64                          Global.beanShell.set("logger", logger);
65                          Global.beanShell.set("generator", Global.generator);
66                          Global.beanShell.set("coefficients", coefficients);
67                          
68                          Object o = Global.beanShell.source(scriptFile);
69                          
70                          boolean b = true;
71                          try{
72                              b = Boolean.parseBoolean(o.toString());
73                          } catch(Exception e){
74                          }
75                          
76                          actor.setAbilityCooldown(this,cooldown);
77                          logger.info("Ability completed");
78                          return b;
79                          
80                      } catch (IOException e) {
81                          e.printStackTrace();
82                          logger.debug(e);
83                      } catch (EvalError e) {
84                          e.printStackTrace();
85                          logger.debug(e);
86                      }
87                  }
88                  else{
89                      RPGMain.printText(true, "Target is too far away to do that.");
90                  }
91              }
92              else{
93                  RPGMain.printText(true, "Target is too close to do that.");
94              }
95          }
96          else{
97              RPGMain.printText(true, "You are not recovered yet from the last time you did that.");
98          }
99          logger.info("Ability failed");
100         return false;
101     }
