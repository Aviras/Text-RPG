307     public void loadWereld(File aFile){
308         try {
309             Document doc = parser.build(aFile);
310             //there is only 1 element, so the root element is the World Map objects
311             //all children are already the actual properties
312             Element root = doc.getRootElement();
313             @SuppressWarnings("unchecked")
314             List<Element> objects = root.getChildren();
315             //i iterates over the actual properties
316             Iterator<Element> i = objects.iterator();
317             String[] heightMap = null;
318             while(i.hasNext()){
319                 Element child = i.next();
320                 if(child.getName().equalsIgnoreCase("heightmap")){
321                     heightMap = initiateWorld(child.getTextTrim());
322                 }
323                 else if(child.getName().equalsIgnoreCase("extra")){
324                     populateWorld(child,heightMap);
325                 }
326             }
327         } catch (JDOMException e) {
328             e.printStackTrace();
329             logger.debug(e);
330         } catch (IOException e) {
331             e.printStackTrace();
332             logger.debug(e);
333         }
334     }
335     public String[] initiateWorld(String s){
336         String[] lines = s.split(System.getProperty("line.separator"));
337         if(lines.length == 1){
338             lines = s.split("\n");
339         }
340         int height = lines.length;
341         int width = lines[0].length();
342         
343         wereld = new Location[width][height];
344         
345         return lines;
346     }
347     public void populateWorld(Element e, String[] heightMap){
348         @SuppressWarnings("unchecked")
349         List<Element> locations = e.getChildren();
350         Iterator<Element> i = locations.iterator();
351         /*for(int j=0;j<wereld.length;j++){
352             for(int k=0;k<wereld[0].length;k++){
353                 wereld[j][k] = new Sea(0,"Default Sea", new int[]{j,k},null);
354             }
355         }*/
356         // populate world with towns & hostileAreas
357         while(i.hasNext()){
358             Element child = i.next();
359             int x = Integer.parseInt(child.getAttributeValue("x"));
360             int y = Integer.parseInt(child.getAttributeValue("y"));
361             if(child.getName().equalsIgnoreCase("town")){
362                 wereld[x][y] = towns.get(Integer.parseInt(child.getTextNormalize()));
363                 towns.get(Integer.parseInt(child.getTextNormalize())).setPosition(x, y);
364             }
365             else if(child.getName().equalsIgnoreCase("hostileArea")){
366                 wereld[x][y] = hostileAreas.get(Integer.parseInt(child.getTextNormalize()));
367                 hostileAreas.get(Integer.parseInt(child.getTextNormalize())).setPosition(x, y);
368             }
369             else if(child.getName().equalsIgnoreCase("sea")){
370                 wereld[x][y] = new Sea(0,"Default Sea", new int[]{x,y},null);
371                 /*try{
372                     //TODO
373                     //TODO make a linear mapping from the in-game map to the drawn map, so it's easy to put the right position
374                     wereld[x][y] = seas.get(Integer.parseInt(child.getTextNormalize()));
375                     seas.get(Integer.parseInt(child.getTextNormalize())).setPosition(x, y);
376                 } catch(NullPointerException exc){
377                 } catch(ArrayIndexOutOfBoundsException exc){
378                 }*/
379             }
380         }
381         // set world level heights
382         for(int j=0;j<heightMap.length;j++){
383             for(int k=0;k<heightMap[0].length();k++){
384                 try{
385                     wereld[k][j].setAltitude(Character.digit(heightMap[j].charAt(k), 10));
386                 }catch(NullPointerException np){
387                 }
388             }
389         }
390     }
