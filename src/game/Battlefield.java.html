63      public void updateBattlefield(HashMap<double[][],Double> covers,int[] newPlayerCoords,ArrayList<int[]> newEnemyCoords){
64          atEnemy = true;
65          //will recieve null when enemy is dead
66          enemyCoords.clear();
67          visibleEnemies.clear();
68          if(newEnemyCoords == null){
69              atEnemy = false;
70          }
71          else{
72              this.covers = covers;
73              playerCoords[0] = newPlayerCoords[0];
74              playerCoords[1] = newPlayerCoords[1];
75              for(int j=0;j<newEnemyCoords.size();j++){
76                  enemyCoords.add(newEnemyCoords.get(j));
77                  visibleEnemies.add(HostileArea.lineOfSight(playerCoords, newEnemyCoords.get(j), covers));
78              }
79          }
80          //System.out.println("Repaint will get called from updateBattlefield. Playercoord: " + playerCoords[0] + ", " + playerCoords[1]);
81          repaint();
82      }
83      public void updateBattlefield(HashMap<double[][],Double> covers, int[] newPlayerCoords,int[] newEnemyCoords,int mobIndex){
84          ArrayList<int[]> dummy = Global.cloneArrayList(enemyCoords);
85          dummy.set(mobIndex, newEnemyCoords);
86          updateBattlefield(covers,newPlayerCoords,dummy);
87      }

89          logger.info("newPlayerCoords: " + newPlayerCoords[0] + "," + newPlayerCoords[1]);
90          ArrayList<int[]> dummy = Global.cloneArrayList(enemyCoords);
91          updateBattlefield(covers,newPlayerCoords,dummy);
92      }

102     public void drawBattlefield(Graphics g){
103         //System.out.println("drawBattlefield gets called");
104         Color[] colors = GameFrameCanvas.dungeonMap.getCurrentColorSet();
105         Color npcColor = GameFrameCanvas.dungeonMap.getNpcColor();
106         Color dungeonColor = GameFrameCanvas.dungeonMap.getDungeonColor();
107         Color enemyColor = GameFrameCanvas.dungeonMap.getEnemyColor();
108         
109         int currentLevel = GameFrameCanvas.dungeonMap.getDungeonLevel();
110         int[][] elevation = GameFrameCanvas.dungeonMap.getElevation();
111         int[] playerPos = GameFrameCanvas.dungeonMap.getPlayerPosition();
112         if(currentLevel == 0){
113             //MIDDLE
114             g.setColor(colors[elevation[playerPos[0]][playerPos[1]]]);
115             setBackground(colors[elevation[playerPos[0]][playerPos[1]]]);
116             //SOUTH
117             try{
118                 g.setColor(colors[elevation[playerPos[0]][playerPos[1]+1]]);
119                 g.fillRect(getWidth()/3, 2*getHeight()/3, getWidth()/3, getHeight()/3);
120             }catch(ArrayIndexOutOfBoundsException e){
121             }
122             //WEST
123             try{
124                 g.setColor(colors[elevation[playerPos[0]-1][playerPos[1]]]);
125                 g.fillRect(0, getHeight()/3, getWidth()/3, getHeight()/3);
126             } catch(ArrayIndexOutOfBoundsException e){
127             }
128             //EAST
129             try{
130                 g.setColor(colors[elevation[playerPos[0]+1][playerPos[1]]]);
131                 g.fillRect(2*getWidth()/3, getHeight()/3, getWidth()/3, getHeight()/3);
132             } catch(ArrayIndexOutOfBoundsException e){
133             }
134             //NORTH
135             try{
136                 g.setColor(colors[elevation[playerPos[0]][playerPos[1]-1]]);
137                 g.fillRect(getWidth()/3, 0, getWidth()/3, getHeight()/3);
138             } catch(ArrayIndexOutOfBoundsException e){
139             }
140             //NORTH EAST
141             try{
142                 g.setColor(colors[elevation[playerPos[0]+1][playerPos[1]-1]]);
143                 g.fillRect(2*getWidth()/3, 0, getWidth()/3, getHeight()/3);
144             } catch(ArrayIndexOutOfBoundsException e){
145             }
146             //NORTH west
147             try{
148                 g.setColor(colors[elevation[playerPos[0]-1][playerPos[1]-1]]);
149                 g.fillRect(0, 0, getWidth()/3, getHeight()/3);
150             } catch(ArrayIndexOutOfBoundsException e){
151             }
152             //SOUTH EAST
153             try{
154                 g.setColor(colors[elevation[playerPos[0]+1][playerPos[1]+1]]);
155                 g.fillRect(2*getWidth()/3, 2*getHeight()/3, getWidth()/3, getHeight()/3);
156             } catch(ArrayIndexOutOfBoundsException e){
157             }
158             //SOUTH WEST
159             try{
160                 g.setColor(colors[elevation[playerPos[0]-1][playerPos[1]+1]]);
161                 g.fillRect(0, 2*getWidth()/3, getWidth()/3, getHeight()/3);
162             } catch(ArrayIndexOutOfBoundsException e){
163             }
164             /*File file;
165             try{
166                 file = new File("Textures/Grass0138_35_S.jpg");
167                 Image background = ImageIO.read(file);
168                 g.drawImage(background, 0, 0, this.getWidth(), this.getHeight(), null);
169             } catch(Exception e){
170                 e.printStackTrace();
171                 logger.error(e);
172             }*/
173         }
174         else{
175             g.setColor(dungeonColor);
176             g.fillRect(0, 0, getWidth(), getHeight());
177         }
178         //COVERS
179         g.setColor(Color.black);
180         Set<double[][]> c = covers.keySet();
181         Iterator<double[][]> coverPos = c.iterator();
182         while(coverPos.hasNext()){
183             // key looks like this:
184             // posX posY
185             // sizX sizY
186             g.setColor(Color.black);
187             double[][] key = coverPos.next();
188             g.fillOval((int)key[0][0] - (int)(Global.coverSize/2*key[1][0]), (int)key[0][1] - (int)(Global.coverSize/2*key[1][1]), (int)(Global.coverSize*key[1][0]), (int)(Global.coverSize*key[1][1]));
189             g.setColor(Color.white);
190             DecimalFormat df = new DecimalFormat("#.##");
191             g.drawString("" + df.format(covers.get(key)),(int)key[0][0] - (int)(Global.coverSize/2*key[1][0]), (int)key[0][1] - (int)(Global.coverSize/2*key[1][1]));
192         }
193         //PLAYER & ENEMIES
194         g.setColor(npcColor);
195         //System.out.println("Player coords at drawBattlefield: " + playerCoords[0] + "," + playerCoords[1]);
196         g.fillOval(playerCoords[0] - Global.playerSize/2, playerCoords[1] - Global.playerSize/2, Global.playerSize, Global.playerSize);
197         if(!hasMovedMax){
198             g.setColor(Color.red);
199             int movementInPixels = (int)(RPGMain.speler.getMovement()/sizeInMeter*getWidth());
200             logger.info("MovementInPixels: " + movementInPixels + "Movement in meters: " + RPGMain.speler.getMovement());
201             g.drawOval(playerCoords[0]-movementInPixels, playerCoords[1] - movementInPixels, 2*movementInPixels, 2*movementInPixels);
202         }
203         for(int j=0;j<enemyCoords.size();j++){
204             int[] enemyCoord = enemyCoords.get(j);
205             if(visibleEnemies.get(j)){
206                 if(j == playerTargetIndex){
207                     g.setColor(Color.GREEN);
208                 }
209                 else{
210                     g.setColor(enemyColor);
211                 }
212                 g.fillOval(enemyCoord[0] - Global.playerSize/2, enemyCoord[1] - Global.playerSize/2, Global.playerSize, Global.playerSize);
213                 g.setColor(Color.black);
214                 g.drawString("" + (j+1), enemyCoord[0] + Global.playerSize/2, enemyCoord[1] - Global.playerSize/2);
215             }
216             else{
217                 g.fillOval(enemyCoord[0], enemyCoord[1], 5, 5);
218             }
219         }
220     }

245     public void mousePressed(MouseEvent arg0) {
246         if(atEnemy && !hasMovedMax){
247             PointerInfo a = MouseInfo.getPointerInfo();
248             Point point = new Point(a.getLocation());
249             SwingUtilities.convertPointFromScreen(point, this);
250             if(point.distance(new Point(playerCoords[0],playerCoords[1])) <= RPGMain.speler.getMovement()/sizeInMeter*getHeight()){
251                 playerCoords[0] = (int)point.getX();
252                 playerCoords[1] = (int)point.getY();
253             }
254         }
