104     class TemperatureChecker extends Thread implements Serializable{
105         
106         /**
107          * 
108          */
109         private static final long serialVersionUID = 1L;
110         double currentTemperature;
111         boolean[] messaged = {false,false,false};
112         int mantleValue,shirtValue,pantsValue;
113         
114         public TemperatureChecker(double c){
115             currentTemperature = c;
116             logger.info("Started temperature check");
117             start();
118         }
119         public void run(){
120             // too cold on chest: less strength, slower -> bigger chance to get hit
121             // too cold on feet: slower movement, damage after a while
122             // too cold on cloak: take damage because not shielded from elements
123             logger.debug("Started run method for Temperature checker");
124             
125             
126             while(RPGMain.speler != null && !RPGMain.speler.checkDood() && RPGMain.speler.getCurrentPosition() == positie){
127                 logger.debug("In while loop of TemperatureChecker");
128                 logger.debug("T: " + currentTemperature);
129                 
130                 checkChanges();
131                 resetClothingLabels();
132                 
133                 logger.debug("Clothing values: " + mantleValue + ", " + shirtValue + ", " + pantsValue);
134                 
135                 /*
136                  * MANTLE
137                  */
138                 // cold
139                 if(7*mantleValue + currentTemperature < 8){
140                     if(!messaged[0]){
141                         GameFrameCanvas.clothingIcons[0].setVisible(true);
142                         GameFrameCanvas.clothingIcons[0].setToolTipText("It's cold out here! The wind isn't helping either. You could use a thicker cloak. You're burning through your food reserves fast.");
143                     }
144                     //TODO
145                     RPGMain.speler.addHunger((int)(Math.abs(7*mantleValue+currentTemperature-8)));
146                     messaged[0] = true;
147                 }
148                 // hot
149                 else if(10*mantleValue + currentTemperature > 30){
150                     if(!messaged[0]){
151                         GameFrameCanvas.clothingIcons[0].setVisible(true);
152                         GameFrameCanvas.clothingIcons[0].setToolTipText("It's hot with that mantle on! Dehydration will kill you if you aren't careful. Better find something to drink soon.");
153                     }
154                     RPGMain.speler.addThirst((int)((10*mantleValue + currentTemperature - 35)));
155                     messaged[0]= true;
156                 }
157                 
158                 /*
159                  * SHIRT
160                  */
161                 // cold
162                 if(7*shirtValue + currentTemperature < 8){
163                     if(!messaged[1]){
164                         GameFrameCanvas.clothingIcons[1].setVisible(true);
165                         GameFrameCanvas.clothingIcons[1].setToolTipText("The cold enters your chest, and feel your arms going numb. Strength continually being reduced by " + Math.abs(7*shirtValue + currentTemperature-8) + "%, and your reflexes are slowed.");
166                     }
167                     // decrease strength and dexterity
168                     //TODO balance numbers
169                     //RPGMain.speler.increaseStat(0, -Math.abs(7*shirtValue + currentTemperature-8)/100*(double)RPGMain.speler.getStrength());
170                     //RPGMain.speler.increaseStat(1, -Math.abs(7*shirtValue + currentTemperature-8)/100*(double)RPGMain.speler.getDexterity());
171                     messaged[1] = true;
172                 }
173                 // hot
174                 else if(15*shirtValue + currentTemperature > 45){
175                     if(!messaged[1]){
176                         GameFrameCanvas.clothingIcons[1].setVisible(true);
177                         GameFrameCanvas.clothingIcons[1].setToolTipText("You're sweating heavily with this shirt, and the armour isn't getting any lighter. You're getting a bit short of breath. Strength continually being reduced by " + (15*shirtValue + currentTemperature - 45) + "%, and your reflexes are slowed");
178                     }
179                     //TODO balance numbers
180                     //RPGMain.speler.increaseStat(0, -(15*shirtValue + currentTemperature - 45)/100*(double)RPGMain.speler.getStrength());
181                     //RPGMain.speler.increaseStat(1, -(15*shirtValue + currentTemperature - 45)/100*(double)RPGMain.speler.getDexterity());
182                     messaged[1] = true;
183                 }
184 
185                 /*
186                  * PANTS
187                  */
188                 // cold
189                 if(7*pantsValue + currentTemperature < 8){
190                     if(!messaged[2]){
191                         GameFrameCanvas.clothingIcons[2].setVisible(true);
192                         GameFrameCanvas.clothingIcons[2].setToolTipText("This cold isn't doing any good for your feet and legs. They're suffering, and it takes more and more effort to take another step. Movement continually being reduced by " + Math.abs(7*pantsValue + currentTemperature-8) + "%.");
193                     }
194                     //TODO balance numbers. Right now: decreases until at 33% of original movement
195                     RPGMain.speler.dimMovement(Math.abs(7*pantsValue + currentTemperature-8)/100);
196                     messaged[2] = true;
197                 }
198                 // hot
199                 else if(15*pantsValue + currentTemperature > 45){
200                     if(!messaged[2]){
201                         GameFrameCanvas.clothingIcons[2].setVisible(true);
202                         GameFrameCanvas.clothingIcons[2].setToolTipText("Your feet are dry, every step is starting to hurt more. Movement continually being reduced by " + (15*pantsValue + currentTemperature - 45) + "%");
203                     }
204                     //TODO balance numbers
205                     RPGMain.speler.dimMovement((15*pantsValue + currentTemperature - 45)/100);
206                     messaged[2] = true;
207                 }
208                 else{
209                     RPGMain.speler.dimMovement(-10);
210                 }
211                 
212                 try{
213                     logger.info("TemperatureChecker sleeping for 10 seconds.");
214                     sleep(10000);
215                 } catch(InterruptedException e){
216                 }
217             }

222         public void checkChanges(){
223             if(RPGMain.speler.getMantle().getWarmth() != mantleValue){
224                 messaged[0] = false;
225             }
226             if(RPGMain.speler.getShirt().getWarmth() != shirtValue){
227                 messaged[1] = false;
228             }           
229             if(RPGMain.speler.getPants().getWarmth() != pantsValue){
230                 messaged[2] = false;
231             }
232             
233             mantleValue = RPGMain.speler.getMantle().getWarmth();
234             shirtValue = RPGMain.speler.getShirt().getWarmth();
235             pantsValue = RPGMain.speler.getPants().getWarmth();
236         }
237         public void resetClothingLabels(){
238             for(int j=0;j<3;j++){
239                 if(!messaged[j]){
240                     JLabel jl = GameFrameCanvas.clothingIcons[j];
241                     jl.setVisible(false);
242                     jl.setToolTipText("");
243                 }
244             }
245         }
