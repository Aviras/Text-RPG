311     public static void moveComponents(JComponent compOne, JComponent compTwo, int heightOne, int time){
312         int xOne = compOne.getX();
313         int xTwo = compTwo.getX();
314         int wOne = compOne.getWidth();
315         int wTwo = compTwo.getWidth();
316         int hOne = compOne.getHeight();
317         int hTwo = compTwo.getHeight();
318         
319         compOne.setVisible(true);
320         compTwo.setVisible(true);
321         
322         layout.putConstraint(n, compTwo, 10, s, compOne);
323         
324         for(int j=0;j<30;j++){
325             
326             int yOne = compOne.getY();
327             int yTwo = compTwo.getY();
328             
329             
330             compOne.setPreferredSize(new Dimension(wOne, yTwo-yOne + j*(heightOne-hOne)/30));
331             compTwo.setPreferredSize(new Dimension(wTwo, hTwo - (yTwo - yOne) - j*(heightOne-hOne)/30));
332             
333             
334             try{
335                 Thread.sleep((int)((double)time/30.0));
336                 System.out.println("Sleeping");
337             } catch(InterruptedException e){
338                 e.printStackTrace();
339             }
340         }

445     public void loadSounds(String fileName){
446         try {
447             AudioInputStream in = AudioSystem.getAudioInputStream(new File(fileName));
448             DataLine.Info info = new DataLine.Info(Clip.class,in.getFormat());
449             Clip clip = (Clip) AudioSystem.getLine(info);
450             clip.open(in);
451             clips.add(clip);
452         } catch (UnsupportedAudioFileException e) {
453         } catch (IOException e) {
454         } catch (LineUnavailableException e) {
455         }
456     }
457     public void playSound(int index,int loop){
458         LOOP = loop;
459         SOUNDINDEX = index;
460         clips.get(index).loop(LOOP);
461     }
