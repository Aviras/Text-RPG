38      @SuppressWarnings("unchecked")
39      public void buildCity(Element el){
40          List<Element> d = el.getChildren();
41          Iterator<Element> i = d.iterator();
42          int maxX = 0,maxY = 0;
43          while(i.hasNext()){
44              Element district = i.next();
45              int x = Integer.parseInt(district.getAttributeValue("x"));
46              int y = Integer.parseInt(district.getAttributeValue("y"));
47              if(x>maxX){
48                  maxX = x;
49              }
50              if(y>maxY){
51                  maxY = y;
52              }
53          }
54          districts = new District[maxX+1][maxY+1];
55          i = d.iterator();
56          while(i.hasNext()){
57              Element district = i.next();
58              int x = Integer.parseInt(district.getAttributeValue("x"));
59              int y = Integer.parseInt(district.getAttributeValue("y"));
60              String name = district.getChildText("name");
61              String description = district.getChildText("description");
62              String qnpcs = district.getChildText("npc");
63              List<Element> locations = district.getChildren("location");
64              districts[x][y] = new District(new int[] {x,y},name,description,qnpcs,locations);
65          }
66      }

601     class District implements Serializable{
602         /**
603          * 
604          */
605         private static final long serialVersionUID = 1L;
606         private String districtName,enterDescription;
607         private ArrayList<DistrictLocation> locations;
608         private ArrayList<Gate> gates;
609         private int[] districtPos;
610         private ArrayList<NPC> presentNPCs;
611         
612         public District(int[] pos, String name, String description, String npcs, List<Element> extra){
613             districtPos = pos;
614             districtName = name;
615             enterDescription = description;
616             locations = new ArrayList<DistrictLocation>();
617             presentNPCs = new ArrayList<NPC>();
618             gates = new ArrayList<Gate>();
619             
620             if(npcs != null){
621                 String[] t = npcs.split(";");
622                 for(String s:t){
623                     try{
624                         presentNPCs.add(Data.NPCs.get(Integer.parseInt(s)));
625                     }catch(NumberFormatException e){
626                     }catch(IndexOutOfBoundsException e){
627                     }
628                 }
629             }
630             
631             Iterator<Element> i = extra.iterator();
632             while(i.hasNext()){
633                 Element el = i.next();
634                 String nameDummy = el.getChildText("name");
635                 String descriptionDummy = el.getChildText("description");
636                 String type = el.getChildText("type");
637                 String itemsPresent = el.getChildText("itemsPresent");
638                 //TODO allow multiple npcs and still separate main npc (innkeeper, shopkeeper)
639                 String npcID = el.getChildText("npcID");
640                 String connectedCities = el.getChildText("connectedCities");
641                 if(type.equalsIgnoreCase("shop")){
642                     locations.add(new Winkel(nameDummy,0,descriptionDummy,itemsPresent,npcID));
643                 }
644                 else if(type.startsWith("gate")){
645                     gates.add(new Gate(nameDummy,descriptionDummy,type));
646                 }
647                 else if(type.equalsIgnoreCase("inn")){
648                     locations.add(new Inn(nameDummy,descriptionDummy,npcID));
649                 }
650                 else if(type.equalsIgnoreCase("library")){
651                     locations.add(new Library(nameDummy,descriptionDummy,itemsPresent));
652                 }
653                 else if(type.equalsIgnoreCase("house")){
654                     locations.add(new House(nameDummy,descriptionDummy,npcID,itemsPresent));
655                 }
656                 else if(type.equalsIgnoreCase("stables")){
657                     locations.add(new Stables(nameDummy,descriptionDummy,npcID,connectedCities));
658                 }
659             }
660         }
661         public void updateEmoValues(){
662             for(NPC n: presentNPCs){
663                 n.addEmotionValue("happy", 0.10*prevHope);
664                 n.addEmotionValue("angry", -0.10*prevHope);
665                 n.addEmotionValue("hopeful", 0.15*prevHope);
666                 n.addEmotionValue("frightened", -0.15*prevHope);
667             }
668         }
669         public void determineNPCMentalState(){
670             HashMap<String,Double> emotions;
671             for(NPC n: presentNPCs){
672                 
673                 emotions = n.getEmotionValues();
674                 double sum = 0;
675                 double random = Global.generator.nextDouble();
676                 for(String s: emotions.keySet()){
677                     sum+=emotions.get(s);
678                     if(random < sum){
679                         n.addEmotionValue(s, 1.0);
680                         break;
681                     }
682                 }
683             }
684         }
685         public boolean hasInn(){
686             for(DistrictLocation d: locations){
687                 if(d instanceof Inn){
688                     return true;
689                 }
690             }
691             return false;
692         }
693         public boolean hasStables(){
694             for(DistrictLocation d: locations){
695                 if(d instanceof Stables){
696                     return true;
697                 }
698             }
699             return false;
700         }
