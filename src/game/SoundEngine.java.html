403         public void run(){
404             while(loop >= 0 && index < fileNames.length){
405                 logger.debug("New cycle of " + fileNames[index]);
406                 playSound();
407                 if(loop < 0){
408                     logger.debug("Incrementing index for " + fileNames[index]);
409                     loop = INITLOOP;
410                     index++;
411                 }
412                 if(!dLine.isOpen()){
413                     logger.debug("Returning because of externally closed line.");
414                     loop = -1;
415                     index++;
416                 }
417                 if(type.equalsIgnoreCase("ambient")){
418                     removeClosed(ambientLines);
419                 }
420                 else if(type.equalsIgnoreCase("combat")){
421                     removeClosed(combatLines);
422                 }
423                 else if(type.equalsIgnoreCase("effects")){
424                     removeClosed(effectsLines);
425                 }
426                 else if(type.equalsIgnoreCase("weather")){
427                     removeClosed(weatherLines);
428                 }
429             }
430             logger.debug("Thread for " + fileNames[index-1] + " has ended.");
431         }
432         private void playSound(){
433             FloatControl volCtrl = null;
434             //Clip clip = null;
435             SourceDataLine line = null;
436 
437             logger.debug("In new playSound method for " + fileNames[index]);
438 
439             if(gain == -100){
440                 logger.debug("Adjusting gain for " + fileNames[index]);
441                 if(type.equalsIgnoreCase("music")){
442                     gain = musicGain;
443                 }
444                 else if(type.equalsIgnoreCase("combat")){
445                     gain = combatGain;
446                 }
447                 else if(type.equalsIgnoreCase("effects")){
448                     gain = effectsGain;
449                 }
450                 else if(type.equalsIgnoreCase("ambient")){
451                     gain = ambientGain;
452                 }
453                 else if(type.equalsIgnoreCase("weather")){
454                     gain = weatherGain;
455                 }
456             }
457 
458             try{
459                 if(musicLine.isOpen() && type.equalsIgnoreCase("music")){
460                     if(!musicLine.isActive()){
461                         musicLine.stop();
462                         musicLine.flush();
463                         musicLine.close();
464                     }
465                     else{
466                         new Fade(musicLine,gain);
467                     }
468                 }
469             } catch(NullPointerException e){
470             }
471 
472             logger.debug("New Line for " + fileNames[index] + " " + line + ", looping " + loop + " times.");
473 
474             try {
475                 //System.out.println(fileNames[index]);
476                 AudioInputStream in = AudioSystem.getAudioInputStream(new File(fileNames[index]));
477                 AudioInputStream din = null;
478                 AudioFormat baseFormat = in.getFormat();
479                 AudioFormat decodedFormat = new AudioFormat(AudioFormat.Encoding.PCM_SIGNED,
480                         baseFormat.getSampleRate(),
481                         16,
482                         baseFormat.getChannels(),
483                         baseFormat.getChannels() * 2,
484                         baseFormat.getSampleRate(),
485                         false);
486                 din = AudioSystem.getAudioInputStream(decodedFormat, in);
487 
488                 line = getLine(decodedFormat);
489                 addLine(line);
490 
491                 volCtrl = (FloatControl)line.getControl(FloatControl.Type.MASTER_GAIN);
492                 volCtrl.setValue(gain);
493 
494                 try{
495                     logger.debug(type + " delay " + delay[index]);
496                     sleep(delay[index]);
497                 } catch(InterruptedException e){
498                     e.printStackTrace();
499                     logger.error(e);
500                 }
501                 if(line.isOpen()){
502                     try{
503                         rawplay(decodedFormat, din,line);
504                         loop--;
505                     } catch(Exception e){
506                         e.printStackTrace();
507                         logger.error(e);
508                     }
509                 }
510                 else{
511                     loop = -1;
512                 }
513                 //close lines
514                 in.close();
515                 din.close();
516             } catch (UnsupportedAudioFileException e1) {
517                 e1.printStackTrace();
518                 logger.error(e1);
519                 loop = -1;
520             } catch (IOException e1) {
521                 e1.printStackTrace();
522                 logger.error(e1);
523                 loop = -1;
524             } catch(LineUnavailableException e){
525                 e.printStackTrace();
526                 logger.error(e);
527                 loop = -1;
528             } catch (ArrayIndexOutOfBoundsException e){
529                 e.printStackTrace();
530                 logger.error(e);
531                 loop = -1;
532             } catch(Exception e){
533                 e.printStackTrace();
534                 logger.error(e);
535                 loop = -1;
536             }
537         }

539         private int rawplay(AudioFormat targetFormat, AudioInputStream din, SourceDataLine line) throws IOException, LineUnavailableException{
540             logger.debug("In Rawplay for " + fileNames[index] + " " + din + " " + line);
541             byte[] data = new byte[4096];
542             if (line != null)
543             {
544                 logger.debug("Flushing and starting lines for " + fileNames[index]);
545                 // Start
546                 line.flush();
547                 line.start();
548                 int nBytesRead = 0, nBytesWritten = 0;
549                 while (nBytesRead != -1)
550                 {
551                     nBytesRead = din.read(data, 0, data.length);
552                     if (nBytesRead != -1){
553                         nBytesWritten = line.write(data, 0, nBytesRead);
554 
555                     }
556                 }
557                 if(loop == 0){
558                     // Stop after 500ms, else line gets closed before audio can play if it's a short sample
559                     try {
560                         sleep(500);
561                     } catch (InterruptedException e) {
562                         e.printStackTrace();
563                     }
564                     logger.debug("Stopping and closing lines for " + fileNames[index]);
565                     line.stop();
566                     line.flush();
567                     din.close();
568                 }
569                 return nBytesWritten;
570             }
571             return 0;
572         }
