12      private static final long serialVersionUID = 1L;

14      private static final int DAY_MS = 1800000;

16      private static HopeSimulator hopeSim;

31      public void run(){
32          while(!RPGMain.speler.checkDood()) {
33              
34              updateHour();
35              System.out.println("Hour " + hour);
36              try{
37                  sleep(DAY_MS/24);
38              } catch(InterruptedException e){
39              }
40              RPGMain.speler.addFitness(-5);
41              RPGMain.speler.addHunger(1);
42              RPGMain.speler.addThirst(2);
43          }
44      }
45      public static void giveDescriptiveHour(){
46          if(hour < 5){
47              RPGMain.printText(true, "It is the middle of the night, and a pitchblack night.");
48          }
49          else if(hour < 8){
50              RPGMain.printText(true, "It is the early morning, and the world is ready to awake.");
51          }
52          else if(hour < 12){
53              RPGMain.printText(true, "It is morning, and the sun is on its way to the south.");
54          }
55          else if(hour < 15){
56              RPGMain.printText(true, "It is noon, and the sun rises high in the sky.");
57          }
58          else if(hour < 19){
59              RPGMain.printText(true, "It is afternoon, and the sun is soon leaving the world again.");
60          }
61          else if(hour < 25){
62              RPGMain.printText(true, "It is evening. The sun has left the world again, and its inhabitants are preparing for the night.");
63          }
64          try {
65              Global.pauseProg(2000);
66          } catch (InterruptedException e) {
67              e.printStackTrace();
68          }
69          
70      }
71      @SuppressWarnings("unused")
72      public static void updateHour(){
73          
74          //regulate hope
75          hopeSim.calculateHope();
76          hopeSim.checkTermination();
77          hopeSim.checkHopeImpact();
78          
79          //regulate weather
80          weatherSim.calcTemperatureChange();
81          weatherSim.calculatePressurePoints();
82          weatherSim.calculatePressurePointMovement();
83          weatherSim.calculateCloudMatterDevelopment();
84          weatherSim.calculateCloudMovement();
85          weatherSim.calculateRainFall();
86          WeatherSimulator.determineWeatherSounds();
87          
88          //dynamically update drawing radius
89          GameFrameCanvas.dungeonMap.repaint();
90          
91          RPGMain.speler.decrFoodFreshness();
92          
93          // change time in GUI text box
94          String am_pm;
95          hour++;
96          hour%=24;
97          if(hour == 0) RPGMain.speler.incrDaysPlayed();
98          if(hour>12) am_pm = "PM";
99          else am_pm = "AM";
100         
101         logger.info("Hour: " + hour);
102         // change textbox to hour%12 + am_pm
103     }
104     public static void rest(int hours){
105         for(int j = 0;j<hours;j++)
106             updateHour();
107     }
