242     public String predictPattern(String actions){
243         int patternNr = intelligence;
244 
245         String[] possibilities = {"head","torso","legs"};
246         double[] occurences = new double[possibilities.length];
247         double efficiency = 0;
248         
249         String prediction = possibilities[(int)(Math.random()*possibilities.length)];
250         
251         try{
252             while(efficiency < (0.3-0.008*actions.length()) && patternNr > 0){
253                 String lastMoves = actions.substring(actions.length()-patternNr);
254                 String consideredMoves = actions.substring(0, actions.length()-patternNr);
255                 
256                 int i = consideredMoves.indexOf(lastMoves);
257                 while(i != -1){
258                     try{
259                         char s = consideredMoves.charAt(i+patternNr);
260                         //TODO check if torso is saved as t
261                         switch(s){
262                         case 'h': occurences[0]+=1; break;
263                         case 't': occurences[1]+=1; break;
264                         case 'l': occurences[2]+=1; break;
265                         }
266                         
267                         i = consideredMoves.indexOf(lastMoves,i+patternNr);
268                         
269                     } catch(IndexOutOfBoundsException exc){
270                         i = -1;
271                     }
272                 }
273                 for(int j=0;j<occurences.length;j++){
274                     if(occurences[j] > 0){
275                         occurences[j]*=1.0/((double)consideredMoves.length()/(double)(patternNr+1));
276                         if(occurences[j] >= efficiency){
277                             efficiency = occurences[j];
278                             prediction = possibilities[j];
279                         }
280                     }
281                 }
282                 patternNr--;
283             }
284         }catch(NullPointerException exc){
285             
286         }catch(IndexOutOfBoundsException exc){
287             
288         }
289         return prediction;
290     }

292         
293         // positions, distances and health perc get calculated, both of enemy and player
294         // width and height of map window get calculated, path gets initiated
295         int[] newPos = new int[2];
296         newPos[0] = myPos[0];

298         double myHealthPerc = (double)HP/(double)maxHP;

300         double distance = new Point(playerPos[0],playerPos[1]).distance(myPos[0],myPos[1]);
301         int width = GameFrameCanvas.battlefield.getWidth();
302         int height = GameFrameCanvas.battlefield.getHeight();
303         ArrayList<int[]> path = new ArrayList<int[]>();
304         
305         // in melee range
306         if(distance*Battlefield.sizeInMeter/width < Data.abilities.get(0).getRange()[1]){
307             switch(intelligence){
308             // least intelligent, will always stay in melee range
309             case 1: break;
310             // will stay in melee until below 33% health, then run away, or if better at ranged
311             case 2: if(myHealthPerc < 0.34 || boog.getStrength() > wapen.getStrength()){
312                         // newPos = oldPos + movement*cos(theta) for x, sin(theta) for y
313                         // min, max so that it stays in playing field
314                         newPos[0] = Math.max(0,Math.min(width-2*Global.playerSize, myPos[0] + (int)(movement*((double)(myPos[0]-playerPos[0])/distance))));
315                         newPos[1] = Math.max(0,Math.min(height-2*Global.playerSize,myPos[1] + (int)(movement*((double)(myPos[1]-playerPos[1])/distance))));
316                     }
317                     break;
318             case 3: if(bestCover > 0 && ((plHealthPerc-myHealthPerc) < 0.2 || (bestCover < 0.6 && boog.getStrength() > wapen.getStrength()))){
319                         newPos[0] = bestCoverPos[0];
320                         newPos[1] = bestCoverPos[1];
321                     }
322                     else if(maxHeightModifier > 0 && (!hasActed || boog.getStrength() > wapen.getStrength())){
323                         newPos[0] = bestHeightPos[0];
324                         newPos[1] = bestHeightPos[1];
325                     }
326                     else if(myHealthPerc < 0.34 || boog.getStrength() > wapen.getStrength()){
327                         newPos[0] = Math.max(0,Math.min(width-2*Global.playerSize, myPos[0] + (int)(movement*((double)(myPos[0]-playerPos[0])/distance))));

329                     }
330             }
331         }
332         // out of melee range
333         else{
334             switch(intelligence){
335             case 1: newPos[0] = myPos[0] - (int)(Math.min(distance - 2*Global.playerSize, movement)*((double)(myPos[0]-playerPos[0])/distance));
336                     newPos[1] = myPos[1] - (int)(Math.min(distance - 2*Global.playerSize, movement)*((double)(myPos[1]-playerPos[1])/distance));
337                     break;
338             case 2: if(myHealthPerc < 0.34){
339                         // newPos = oldPos + movement*cos(theta) for x, sin(theta) for y
340                         // min, max so that it stays in playing field
341                         newPos[0] = Math.max(0,Math.min(width-2*Global.playerSize, myPos[0] + (int)(movement*((double)(myPos[0]-playerPos[0])/distance))));
342                         newPos[1] = Math.max(0,Math.min(height-2*Global.playerSize,myPos[1] + (int)(movement*((double)(myPos[1]-playerPos[1])/distance))));
343                     }
344                     else{
345                         newPos[0] = myPos[0] - (int)(Math.min(distance - 2*Global.playerSize, movement)*((double)(myPos[0]-playerPos[0])/distance));
346                         newPos[1] = myPos[1] - (int)(Math.min(distance - 2*Global.playerSize, movement)*((double)(myPos[1]-playerPos[1])/distance));
347                     }
348                     break;
349                     //TODO next cases
350             }
351         }
352         path = Global.calculatePath(myPos, playerPos, (int)(movement*GameFrameCanvas.battlefield.getWidth()/Battlefield.sizeInMeter), covers,true);
353         for(int j=0;j<path.size();j++){
354             GameFrameCanvas.battlefield.updateBattlefield(covers, playerPos, path.get(j),mobIndex);
355             try{
356                 Thread.sleep(1000/24);
357             } catch(InterruptedException exc){
358                 
359             }
360             if(j == path.size()-1){
361                 return path.get(path.size()-1);
362             }
363         }
364         return myPos;
