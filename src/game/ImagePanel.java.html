19      public ImagePanel(Dimension dim){
20          setLayout(null);

25          try{

40          String[] idComponents = id.split(" ");
41          System.out.println(id);
42          try{
43              if(idComponents[0].equalsIgnoreCase("hostileArea"))
44                  image = ImageIO.read(new File("Images/" + Data.hostileAreas.get(Integer.parseInt(idComponents[1])).getImage()));
45              else if(idComponents[0].equalsIgnoreCase("town"))
46                  image = ImageIO.read(new File("Images/" + Data.towns.get(Integer.parseInt(idComponents[1])).getImage()));
47          } catch(IOException e){
48              e.printStackTrace();
49              logger.debug(e);
50          }
51          repaint();
52      }
