39      public Weatherplots(String title, double[][] data){
40          super(title);
41          DefaultXYZDataset dataset = new DefaultXYZDataset();
42          double[] xvalues = new double[data.length*data[0].length];
43          double[] yvalues = new double[data.length*data[0].length];
44          double[] zvalues = new double[data.length*data[0].length];
45          int i = 0;
46          for(int j=0;j<xvalues.length;j++){
47              if(j%data[0].length == 0 && j!= 0){
48                  i++;
49              }
50              xvalues[j] = i;
51          }
52          for(int j=0;j<yvalues.length;j++){
53              yvalues[j] = j%data[0].length;
54          }
55          for(int j=0;j<zvalues.length;j++){
56              zvalues[j] = data[(int)xvalues[j]][(int)yvalues[j]];
57          }
58          
59          dataset.addSeries("Temperature 1", new double[][]{xvalues,yvalues,zvalues});
60          JPanel chartPanel = createContourPanel(dataset);
61          chartPanel.setPreferredSize(new Dimension(500,500));
62          setContentPane(chartPanel);
63      }
64      public Weatherplots(String title, HashMap<int[],double[]> data){

75          JPanel chartPanel = createVectorPanel(dataset);
76          chartPanel.setPreferredSize(new Dimension(500,500));
77          setContentPane(chartPanel);
78      }

104     private static JFreeChart createChart(VectorXYDataset vectorxydataset)   
105     {   
106         NumberAxis numberaxis = new NumberAxis("X");   
107         numberaxis.setStandardTickUnits(NumberAxis.createIntegerTickUnits());   
108         numberaxis.setLowerMargin(0.01D);   
109         numberaxis.setUpperMargin(0.01D);   
110         numberaxis.setAutoRangeIncludesZero(false);   
111         NumberAxis numberaxis1 = new NumberAxis("Y");   
112         numberaxis1.setStandardTickUnits(NumberAxis.createIntegerTickUnits());   
113         numberaxis1.setLowerMargin(0.01D);   
114         numberaxis1.setUpperMargin(0.01D);   
115         numberaxis1.setAutoRangeIncludesZero(false);   
116         VectorRenderer vectorrenderer = new VectorRenderer();   
117         vectorrenderer.setSeriesPaint(0, Color.blue);   
118         XYPlot xyplot = new XYPlot(vectorxydataset, numberaxis, numberaxis1, vectorrenderer);  
119         xyplot.getRangeAxis().setInverted(true);
120         xyplot.setBackgroundPaint(Color.lightGray);   
121         xyplot.setDomainGridlinePaint(Color.white);   
122         xyplot.setRangeGridlinePaint(Color.white);   
123         xyplot.setAxisOffset(new RectangleInsets(5D, 5D, 5D, 5D));   
124         xyplot.setOutlinePaint(Color.black);   
125         JFreeChart jfreechart = new JFreeChart("Wind Speeds", xyplot);   
126         jfreechart.setBackgroundPaint(Color.white);   
127         return jfreechart;   
128     }

130     private static JFreeChart createChart(XYDataset dataset){
131         JFreeChart chart = ChartFactory.createScatterPlot(
132                 "Cloud Positions",                  // chart title
133                 "X",                      // x axis label
134                 "Y",                      // y axis label
135                 dataset,                  // data
136                 PlotOrientation.VERTICAL,
137                 true,                     // include legend
138                 true,                     // tooltips
139                 false                     // urls
140             );
141         XYPlot plot = (XYPlot) chart.getPlot();
142         plot.getRangeAxis().setInverted(true);
143         XYLineAndShapeRenderer renderer = new XYLineAndShapeRenderer();
144         renderer.setSeriesLinesVisible(0, false);
145         plot.setRenderer(renderer);
146         return chart;
147     }
