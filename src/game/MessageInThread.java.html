24      public MessageInThread(Socket client){
25          this.client = client;
26          Global.online = true;

29      public void run(){
30          try{
31              DataInputStream din = new DataInputStream(client.getInputStream());
32              while(true){
33                  String message = din.readUTF();
34                  // check for data
35                  try{
36                      if(!checkData(message)){// data doesn't get shown
37                          JOptionPane.showMessageDialog(null, message,"Action",JOptionPane.INFORMATION_MESSAGE);
38                      }
39                  } catch(InterruptedException e){}
40              }
41          } catch(SocketException se){
42              RPGMain.printText(true,"Server went offline.");
43          } catch( EOFException ie ) {
44              //This doesn't need an error message
45          } catch( IOException ie ) {
46              //This does; tell the world!
47              ie.printStackTrace();
48          } finally {
49              //The connection is closed for one reason or another,
50              //so have the server dealing with it
51              ConnectThread.removeConnection( client );
52          }
53      }
