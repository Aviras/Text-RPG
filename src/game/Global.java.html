82       * @param repeated Separate text pieces must be marked with a repeated String
83       *                 with an integer. ex.: "HISTORY 1" \n 'Dialog text' "HISTORY 2"
84       * @param optionalAnswers If a dialog requests user input. UNDER CONSTRUCTION

87      public static String makeDialog(File aFile,String[] path){
88          SAXBuilder parser = new SAXBuilder();
89          StringBuilder sb = new StringBuilder();
90          try {
91              Document doc = parser.build(aFile);
92              Element root = doc.getRootElement();
93              Element el = root;
94              for(int j=0;j<path.length;j++){
95                  String s = path[j];
96                  String[] t = s.split(",");
97                  try{
98                      String[] u = t[1].split(":");
99                      @SuppressWarnings("unchecked")
100                     List<Element> children = el.getChildren(t[0]);
101                     //iterate over children to see if attribute value fits
102                     for(Element e:children){
103                         if(e.getAttributeValue(u[0]).equalsIgnoreCase(u[1])){
104                             el = e;
105                         }
106                     }
107                 }catch(Exception e){
108                     el = el.getChild(s);
109                     System.out.println(t[0]);
110                 }
111             }
112             List<?> dialog = el.getChildren();
113             Iterator<?> i = dialog.iterator();
114             while(i.hasNext()){
115                 Element d = (Element)i.next();
116                 // get the NPC name, if there is one
117                 String name = d.getAttributeValue("name");
118                 int ms;
119                 try{
120                     // get the specified delay
121                     ms = Integer.parseInt(d.getAttributeValue("delay"));
122                 } catch(NumberFormatException e){ ms = 0;}
123                 // print the NPC name if there was one
124                 if(name != null) RPGMain.printText(false, name + ": ");
125                 // check message for player name or race and change it accordingly
126                 String message = d.getTextNormalize();
127                 message = message.replaceAll("playerName", RPGMain.speler.getName());
128                 message = message.replaceAll("playerRace", RPGMain.speler.getRace());
129                 RPGMain.printText(true,message);
130                 sb.append(message);
131                 // pause program for specified delay
132                 if(ms == 0) pauseProg();
133                 else Thread.sleep(ms);
134             }
135             if(dialog.isEmpty()){
136                 RPGMain.printText(true, el.getTextTrim());
137                 sb.append(el.getTextTrim());
138             }
139             return sb.toString();
140         } catch (JDOMException e) {
141             e.printStackTrace();
142         } catch (IOException e) {
143             e.printStackTrace();
144             RPGMain.printText(true, "File " + aFile.getAbsolutePath() + " not found!");
145         } catch (InterruptedException e) {
146             e.printStackTrace();
147         }
148         return null;

151     public static ArrayList<int[]> calculatePath(int[] startPos, int[] endPos, int movement, HashMap<double[][],Double> covers, boolean withinMelee){
152         ArrayList<int[]> startPath = new ArrayList<int[]>();
153         // ArrayList for partial path that is first in the queue
154         ArrayList<int[]> firstInQueue = new ArrayList<int[]>();
155         startPath.add(startPos);
156         queue.clear();
157         queue.put(0.0,startPath);
158         
159         final int stepSize = 3;
160         
161         logger.info("Entered calculatePath. startPos: " + startPos[0] + "," + startPos[1] + "; endPos: " + endPos[0] + "," + endPos[1] + "; Movement: " + movement + ".");
162         
163         while(!queue.isEmpty()){
164             //get the first in queue
165             firstInQueue = queue.get(queue.firstKey());
166             
167             logger.info("First in Queue: " + firstInQueue.get(firstInQueue.size()-1)[0] + "," + firstInQueue.get(firstInQueue.size()-1)[1] + ". Movement:" + movement);
168             //get the last coord pair of the first partial path in the queue
169             int[] lastPos = firstInQueue.get(firstInQueue.size()-1);
170             //remove the first partial path from the queue
171             queue.remove(queue.firstKey());
172             
173             // checks around the current position: -stepSize,0,+stepSize for both j and k
174             for(int j=-stepSize;j<=stepSize;j=j+stepSize){
175                 for(int k=-stepSize;k<=stepSize;k=k+stepSize){
176                     if(k == 0 && j==0){
177                         // do nothing for current position
178                     }
179                     else{
180                         // add the new coord pair to the partial path of the first in queue
181                         firstInQueue.add(new int[] {lastPos[0] + k,lastPos[1] + j});
182                         // check if the path is valid, and add it to the queue
183                         addPath(firstInQueue,startPos,endPos,covers,stepSize);
184                         // remove the last coord pair, and repeat with the next around it
185                         firstInQueue.remove(firstInQueue.size()-1);
186                     }
187                 }
188             }
189             
190             logger.debug("Queue size after creation of new paths:" + queue.size());
191             
192             // Because of the use of a TreeMap, it's constantly sorted
193             ArrayList<Double> toDelete = new ArrayList<Double>();
194             Set<Double> keys = queue.keySet();
195             Iterator<Double> i = keys.iterator();
196             try{
197                 // iterate over all paths in the queue
198                 while(i.hasNext()){
199                     Double d = i.next();
200                     ArrayList<int[]> p = queue.get(d);
201                     keys = queue.keySet();
202                     // get the last coord pair of path p
203                     int[] k = p.get(p.size()-1);
204                     //iterator over all other paths in the queue
205                     Iterator<Double> l = keys.iterator();
206                     while(l.hasNext()){
207                         Double m = l.next();
208                         ArrayList<int[]> a = queue.get(m);
209                         if(!p.equals(a)){
210                             //check if path a has overlap with p, and if it has and p has already moved more to get to that point than in path a, remove p
211                             for(int j=1;j<a.size();j++){
212                                 int[] q = a.get(j);
213                                 // effective distance travelled is (j-1)*5
214                                 if(q[0] == k[0] && q[1] == k[1] && (p.size()-1)*stepSize > (j-1)*stepSize){
215                                     // use arrayList to delete afterwards to avoid concurrentmodificationexception
216                                     toDelete.add(d);
217                                     break;
218                                 }
219                             }
220                         }
221                     }
222                 }
223                 for(Double d: toDelete){
224                     queue.remove(d);
225                 }
226                 toDelete.clear();
227             }catch(Exception e){
228                 e.printStackTrace();
229                 logger.error("Error at calculate path. " , e);
230             }
231             //System.out.println("Past redundancy checking");
232             ArrayList<int[]> bestPath = queue.get(queue.firstKey());
233             int[] currentPos = bestPath.get(bestPath.size()-1);
234 
235             if(getTraveledDistance(bestPath) >= movement || (withinMelee && new Point(endPos[0],endPos[1]).distance(currentPos[0],currentPos[1]) < Global.playerSize)){
236                 System.out.println("Goal reached. Movement: " + movement + " distance: " + new Point(startPos[0],startPos[1]).distance(currentPos[0],currentPos[1]) + " traveled: " + getTraveledDistance(bestPath));
237                 return bestPath;
238             }
239         }
240         
241         return null;
242     }

244     public static void addPath(ArrayList<int[]> path, int[] startPos, int[] endPos,HashMap<double[][],Double> covers,int stepSize){
245         int width = GameFrameCanvas.battlefield.getWidth();
246         int height = GameFrameCanvas.battlefield.getHeight();
247         
248         // get the last coord pair in the partial path
249         int[] newPos = path.get(path.size()-1);
250         // check that it is still within the window
251         if(newPos[0] > 0 && newPos[0] < width && newPos[1] > 0 && newPos[1] < height){
252             Iterator<double[][]> coverPos = covers.keySet().iterator();
253             // check if it isn't in cover
254             while(coverPos.hasNext()){
255                 double[][] pos = coverPos.next();
256                 // covers are treated as ellipses
257                 // get the value of the cosine for the angle made between the center of the cover and the new coord, and the x-axis
258                 double cosTheta = (double)(newPos[0]-pos[0][0])/Math.sqrt(Math.pow(newPos[0] - pos[0][0], 2) + Math.pow(newPos[1] - pos[0][1], 2));
259                 double radius = 1.0/Math.sqrt(Math.pow(cosTheta/(Global.coverSize/2*pos[1][0]), 2) + (1-cosTheta*cosTheta)/Math.pow(Global.coverSize/2*pos[1][1],2));
260                 if(new Point(newPos[0],newPos[1]).distance(pos[0][0],pos[0][1]) <= (radius+Global.playerSize/2)){
261                     return;
262                 }
263             }
264             // check for loops
265             for(int j=0;j<path.size()-1;j++){
266                 int[] p = path.get(j);
267                 // check for overlap between a coord, and all the ones following it
268                 for(int k = j+1;k<path.size();k++){
269                     int[] l = path.get(k);
270                     
271                     // if there is overlap, don't add the path to the queue
272                     if(p[0] == l[0] && p[1] == l[1]){
273                         return;
274                     }
275                     // if there is no overlap, and the whole partial path is checked, add the path to the queue
276                     else if((p[0] != l[0] || p[1] != l[1]) && k == (path.size()-1) && j == (path.size()-2)){
277                         try{
278                             queue.put(0.5*getTraveledDistance(path) + new Point(newPos[0],newPos[1]).distance(endPos[0],endPos[1]),cloneArrayList(path));
279                         } catch(Exception e){
280                             e.printStackTrace();
281                             logger.error(e);
282                         }
283                     }
284                 }
285             }
286         }
287         else{
288             return;
289         }

291     public static ArrayList<int[]> cloneArrayList(ArrayList<int[]> a){
292         ArrayList<int[]> b = new ArrayList<int[]>(a.size());
293         for(int[] o:a){
294             b.add(o.clone());
295         }
296         return b;

298     public static double getTraveledDistance(ArrayList<int[]> a){
299         double distance = 0;
300         for(int j=0;j<a.size()-1;j++){
301             distance+= new Point(a.get(j)[0],a.get(j)[1]).distance(a.get(j+1)[0],a.get(j+1)[1]);
302         }
303         return distance;
304     }
