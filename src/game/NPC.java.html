84      private void prepareEmoValues(){
85          emotionValues = new HashMap<String,Double>();
86          emotionValues.put("happy",1.0/NUMBEREMOTIONS);
87          emotionValues.put("angry",1.0/NUMBEREMOTIONS);
88          emotionValues.put("hopeful",1.0/NUMBEREMOTIONS);
89          emotionValues.put("frightened",1.0/NUMBEREMOTIONS);
90          addEmotionValue("happy",0.5*Math.random());
91          addEmotionValue("angry",0.5*Math.random());
92          addEmotionValue("hopeful",0.5*Math.random());
93          addEmotionValue("frightened",0.5*Math.random());

173     public String getMentalState(){
174         String mentalState = null;
175         double max = 0;
176         for(String s: emotionValues.keySet()){
177             if(emotionValues.get(s) > max){
178                 max = emotionValues.get(s);
179                 mentalState = s;
180             }
181         }
182         return mentalState;
183     }
184     public void addEmotionValue(String emotion, double value){
185 
186         String oldMentalState = getMentalState();
187 
188         emotionValues.put(emotion, Math.max(Math.min(emotionValues.get(emotion)+value,0.75),0));
189 
190         double sum = 0;
191         for(String s: emotionValues.keySet()){
192             sum+=emotionValues.get(s);
193         }
194         if(sum != 1){
195             for(String s: emotionValues.keySet()){
196                 emotionValues.put(s, emotionValues.get(s)/sum);
197             }
198         }
199 
200         String newMentalState = getMentalState();
201         if(!newMentalState.equalsIgnoreCase(oldMentalState) && isTalking){
202             if(newMentalState.equalsIgnoreCase("happy")){
203                 RPGMain.printText(true, name + " is pleased with what you said.","green");
204             }
205             else if(newMentalState.equalsIgnoreCase("hopeful")){
206                 RPGMain.printText(true, name + " seems to have found new hope because of what you said.","green");
207             }
208             else if(newMentalState.equalsIgnoreCase("angry")){
209                 RPGMain.printText(true, name + " is not pleased with what you said, and appears angered.","red");
210             }
211             else if(newMentalState.equalsIgnoreCase("frightened")){
212                 RPGMain.printText(true, name + " appears frightened by what you said.","red");
213             }
214         }
215     }
