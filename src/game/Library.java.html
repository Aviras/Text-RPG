31      public Library(String name, String description, String bookIDs, String npcIDs){
32          super(name,description);
33          String[] parts = bookIDs.split(";");
34          for(String s:parts){
35              try{
36                  IDs.add(Integer.parseInt(s));
37              }catch(NumberFormatException e){
38                  continue;
39              }
40          }
41          npcs = new ArrayList<NPC>();
42          parts = npcIDs.split(";");
43          for(String s: parts){
44              try{
45                  npcs.add(Data.NPCs.get(Integer.parseInt(s)));
46              }catch(NumberFormatException e){
47                  continue;
48              }
49          }
50      }

52      public void enter() throws InterruptedException{
53          RPGMain.printText(true, description);
54          Global.pauseProg();
55          if(!npcs.isEmpty()){
56              while(true){
57                  for(NPC n: npcs){
58                      RPGMain.printText(true, new String[]{"* ", "Talk to ", n.getName()}, new String[]{"regular","bold","regular"});
59                  }
60                  RPGMain.printText(false, new String[]{"* ", "Read ", " a book\n* Cancel\n>"}, new String[]{"regular","bold","regular"});
61                  
62                  String input = RPGMain.waitForMessage().toLowerCase();
63                  if(input.startsWith("talk to")){
64                      String name = RPGMain.upperCaseSingle(input.split(" ")[2],0);
65                      
66                      for(NPC n: npcs){
67                          if(n.getName().equalsIgnoreCase(name)){
68                              logger.debug("talking to " + name);
69                              n.talk();
70                              break;
71                          }
72                      }
73                  }
74                  else if(input.equalsIgnoreCase("read")){
75                      readBooks();
76                  }
77                  else if(input.equalsIgnoreCase("cancel")){
78                      RPGMain.printText(true, "You walk out of the " + name + ".");
79                      break;
80                  }
81              }
82          }
83          else{
84              readBooks();
85          }
86      }

88      public void readBooks(){
89          while(true){
90              RPGMain.printText(false, "What book do you want to read?");
91              ArrayList<String> titles = new ArrayList<String>();
92              for(int j=0;j<IDs.size();j++){
93                  int i = IDs.get(j);
94                  String[] path = {"book,id:" + i,"name"};
95                  String title = Global.rwtext.getContent(bookFile, path);
96                  path[1] = "author";
97                  String author = Global.rwtext.getContent(bookFile, path);
98                  titles.add(title + " by " + author);
99                  RPGMain.printText(false, "\n" + (j+1) + ": " + titles.get(j));
100             }
101             RPGMain.printText(false, "\n" + (titles.size()+1) + ": Go back" + "\n>");
102             int choice = 0;
103             try{
104                 choice = Integer.parseInt(RPGMain.waitForMessage())-1;
105             }catch(NumberFormatException e){
106                 continue;
107             } catch (InterruptedException e) {
108                 e.printStackTrace();
109                 continue;
110             }
111             if(choice == titles.size()){
112                 break;
113             }
114             else if(choice > titles.size() || choice < 0){
115                 RPGMain.printText(true, "Not a valid option.");
116             }
117             else{
118                 String[] path = {"book,id:" + IDs.get(choice), "Content"};
119                 Global.makeDialog(bookFile, path);
120                 String[] logbookExtra = {path[0],"Log","path"};
121                 try{
122                     String logbookPath = Global.rwtext.getContent(bookFile, logbookExtra);
123                     logbookExtra[2] = "summary";
124                     String[] content = Global.rwtext.getContent(bookFile, logbookExtra).split(";");
125                     Logbook.addContent(logbookPath, Integer.parseInt(content[0]), content[1]);
126                 } catch(ArrayIndexOutOfBoundsException e){
127                     logger.error("Missing logbook data in book with ID " + IDs.get(choice),e);
128                     System.err.println("Missing data in book");
129                 } catch(NumberFormatException e){
130                     logger.error("Missing logbook data in book with ID " + IDs.get(choice),e);
131                     System.err.println("Missing data in book");
132                 } catch(Exception e){
133                     e.printStackTrace();
134                     logger.error(e);
135                 }
136                 continue;
137             }
138         }
139     }
