49      public void calculateCost(int cityID){
50          //modified A*
51          //use a heuristic as sum of height difference squared + distance between two points + danger
52          
53          int[] startPos = RPGMain.speler.getCurrentPosition();
54          int[] endPos = Data.towns.get(cityID).getPositie();
55          
56          int[][] heightMap = new int[Data.wereld.length][Data.wereld[0].length];
57          
58          for(int j=0;j<Data.wereld.length;j++){
59              for(int k=0;k<Data.wereld[0].length;k++){
60                  try{
61                      heightMap[j][k] = Data.wereld[j][k].getAltitude();
62                  } catch(NullPointerException e){
63                      continue;
64                  }
65              }
66          }
67          
68          ArrayList<int[]> startPath = new ArrayList<int[]>();
69          // ArrayList for partial path that is first in the queue
70          ArrayList<int[]> firstInQueue = new ArrayList<int[]>();
71          startPath.add(startPos);
72          queue.clear();
73          queue.put(0.0,startPath);
74          
75          int stepSize = 1;
76          
77          try{
78          while(!queue.isEmpty()){
79              //get the first in queue
80              firstInQueue = queue.get(queue.firstKey());
81              //get the last coord pair of the first partial path in the queue
82              int[] lastPos = firstInQueue.get(firstInQueue.size()-1);
83              //remove the first partial path from the queue
84              queue.remove(queue.firstKey());
85              
86              // checks around the current position: -stepSize,0,+stepSize for both j and k
87              for(int j=-stepSize;j<=stepSize;j=j+stepSize){
88                  for(int k=-stepSize;k<=stepSize;k=k+stepSize){
89                      if((k == 0 && j==0) || k*j != 0){
90                          // do nothing for current position, and don't consider diagonal movements
91                      }
92                      else{
93                          // add the new coord pair to the partial path of the first in queue
94                          firstInQueue.add(new int[] {lastPos[0] + k,lastPos[1] + j});
95                          // check if the path is valid, and add it to the queue
96                          addPath(firstInQueue,startPos,endPos,heightMap,1);
97                          // remove the last coord pair, and repeat with the next around it
98                          firstInQueue.remove(firstInQueue.size()-1);
99                      }
100                 }
101             }
102             
103             // Because of the use of a TreeMap, it's constantly sorted
104             ArrayList<Double> toDelete = new ArrayList<Double>();
105             Set<Double> keys = queue.keySet();
106             Iterator<Double> i = keys.iterator();
107             try{
108                 // iterate over all paths in the queue
109                 while(i.hasNext()){
110                     Double d = i.next();
111                     ArrayList<int[]> p = queue.get(d);
112                     keys = queue.keySet();
113                     // get the last coord pair of path p
114                     int[] k = p.get(p.size()-1);
115                     //iterator over all other paths in the queue
116                     Iterator<Double> l = keys.iterator();
117                     while(l.hasNext()){
118                         Double m = l.next();
119                         ArrayList<int[]> a = queue.get(m);
120                         if(!p.equals(a)){
121                             //check if path a has overlap with p, and if it has and p has already moved more to get to that point than in path a, remove p
122                             for(int j=1;j<a.size();j++){
123                                 int[] q = a.get(j);
124                                 // effective distance travelled is (j-1)*5
125                                 if(q[0] == k[0] && q[1] == k[1] && (p.size()-1)> (j-1)*stepSize){
126                                     // use arrayList to delete afterwards to avoid concurrentmodificationexception
127                                     toDelete.add(d);
128                                     break;
129                                 }
130                             }
131                         }
132                     }
133                 }
134                 for(Double d: toDelete){
135                     System.out.println("Removing path " + d);
136                     queue.remove(d);
137                 }
138                 toDelete.clear();
139             }catch(Exception e){
140                 e.printStackTrace();
141                 logger.error("Error at calculate path. " , e);
142             }
143             ArrayList<int[]> bestPath = queue.get(queue.firstKey());
144             int[] lp = bestPath.get(bestPath.size()-1);
145             
146             if(lp[0] == endPos[0] && lp[1] == endPos[1]){
147                 logger.debug("Bestpath found");
148                 for(int[] in: bestPath){
149                     System.out.print(in[0] + "," + in[1] + "->");
150                 }
151                 break;
152             }
153 
154         }
155         bestPaths.put(cityID, queue.get(queue.firstKey()));
156         
157         double cost = queue.firstKey();
158         
159         connectedCities[cityID][1] = (int)cost;
160         
161         } catch(Exception e){
162             e.printStackTrace();
163             logger.error(e);
164         }

167     public static void addPath(ArrayList<int[]> path, int[] startPos, int[] endPos, int[][] heightMap, int stepSize){
168         
169         try{
170         // get the last coord pair in the partial path
171         int[] newPos = path.get(path.size()-1);
172         
173         // check that it is still within the window
174         if(newPos[0] < heightMap.length && newPos[0] >= 0 && newPos[1] >= 0 && newPos[1] < heightMap[0].length){
175             //sum of height differences + distance between two points + danger
176             double heuristic = 0;
177             // check for loops
178             for(int j=0;j<path.size();j++){
179                 int[] p = path.get(j);
180                 
181                 try{
182                     //Great height differences means extra distance, and heavier burden
183                     heuristic+=Math.abs(heightMap[p[0]][p[1]] - heightMap[path.get(j+1)[0]][path.get(j+1)[1]]);
184                 } catch(IndexOutOfBoundsException e){
185                     logger.error(e);
186                     break;
187                 }
188                 // check for overlap between a coord, and all the ones following it
189                 for(int k = j+1;k<path.size();k++){
190                     int[] l = path.get(k);
191                     
192                     if(Data.wereld[l[0]][l[1]] instanceof HostileArea){
193                         //more dangerous to go through hostile areas than through towns
194                         heuristic+=3;
195                     }
196                     else if(Data.wereld[l[0]][l[1]] instanceof Town){
197                     }
198                     else{
199                         return;
200                     }
201                     
202                     // if there is overlap, don't add the path to the queue
203                     if(p[0] == l[0] && p[1] == l[1]){
204                         return;
205                     }
206                     // if there is no overlap, and the whole partial path is checked, add the path to the queue
207                     else if((p[0] != l[0] || p[1] != l[1]) && k == (path.size()-1) && j == (path.size()-2)){
208                         try{
209                             //add the distances to the heuristic
210                             heuristic+=getTraveledDistance(path) + Point.distance(newPos[0], newPos[1], endPos[0], endPos[1]);
211                             queue.put(heuristic,cloneArrayList(path));
212                             System.err.println("Added path to queue ending on " + newPos[0] + "," + newPos[1]);
213                         } catch(Exception e){
214                             e.printStackTrace();
215                             logger.error(e);
216                         }
217                     }
218                 }
219             }
220         }
221         else{
222             // don't add anything to the queue
223             return;
224         }
225         } catch(Exception e){
226             e.printStackTrace();
227             logger.error(e);
228         }

230     public static ArrayList<int[]> cloneArrayList(ArrayList<int[]> a){
231         ArrayList<int[]> b = new ArrayList<int[]>(a.size());
232         for(int[] o:a){
233             b.add(o.clone());
234         }
235         return b;
236     }
237     public static double getTraveledDistance(ArrayList<int[]> a){
238         double distance = 0;
239         for(int j=0;j<a.size()-1;j++){
240             distance+= new Point(a.get(j)[0],a.get(j)[1]).distance(a.get(j+1)[0],a.get(j+1)[1]);
241         }
242         return distance;
243     }
