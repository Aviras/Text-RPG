108     public void initCityMap(District[][] districts, int[] playerPos){
109         width = districts.length;
110         height = 0;
111         for(int j=0;j<width;j++){
112             try{
113                 if(height < districts[j].length){

116             } catch(NullPointerException e){
117                 continue;
118             }
119         }
120         
121         this.districts = new int[width][height];
122         
123         logger.debug("width: " + width + " height: " + height);
124         
125         for(int j=0;j<width;j++){
126             for(int k=0;k<height;k++){
127                 if(districts[j][k] != null){
128                     this.districts[j][k] = 1;
129                     if(playerPos[0] == j && playerPos[1] == k){
130                         this.districts[j][k] = 2;

132                 }
133             }
134         }
135         drawCity = true;

138     public void initDungeonMap(HashMap<Integer,DungeonRoom[][]> dungeon, int[] positie){
139         drawCity = false;
140         width = dungeon.get(0).length;
141         height = dungeon.get(0)[0].length;
142         elevation = new HashMap<Integer,int[][]>();
143         visible = new int[width][height];
144         enemies.clear();
145         npcs.clear();
146         water.clear();
147         events.clear();
148         for(int level: dungeon.keySet()){
149             elevation.put(level, new int[width][height]);
150             for(int k=0;k<height;k++){
151                 for(int j=0;j<width;j++){
152                     try{
153                         elevation.get(level)[j][k] = dungeon.get(level)[j][k].getHeight();
154                         if(dungeon.get(level)[j][k].hasAliveEnemies()){
155                             enemies.add(new int[] {j,k,level});
156                         }
157                         if(dungeon.get(level)[j][k].getEvent() != null && dungeon.get(level)[j][k].getEvent().getNPC() != null){
158                             npcs.add(new int[] {j,k,level});
159                         }
160                         if(dungeon.get(level)[j][k].getTerrainType().equalsIgnoreCase("water")){
161                             water.add(new int[] {j,k,level});
162                         }
163                         try{
164                             if(dungeon.get(level)[j][k].getEvent() != null && !dungeon.get(level)[j][k].getEvent().getType().equalsIgnoreCase("lost")
165                                     && !dungeon.get(level)[j][k].getEvent().getType().equalsIgnoreCase("trap")){
166                                 events.add(new int[] {j,k,level});
167                             }
168                         } catch(NullPointerException e){
169                             //System.err.println("Nullpointer at " + j + "," + k + "," + level);
170                         }
171                     } catch(NullPointerException e){
172                         //walls of dungeon are high so you can never look around the corner
173                         elevation.get(level)[j][k] = 10;
174                         //System.out.print("wall ");
175                     }
176                 }
177             }
178         }

182             radius = 4;
183             colors = colorsForest;

187             colors = colorsForest;
188         }
189         else if(type.equalsIgnoreCase("plains")){
190             radius = 6;
191             colors = colorsForest;
192         }

214     public void updatePlayerPos(int[] pos, int[] direction,int currentLevel){
215         playerPos = pos;
216         this.currentLevel = currentLevel;
217         int newRadius = Math.max(drawRadius+1, 4);
218         //+/-1 is to prevent system from bugging out when you do an event in this square and go in the opposite direction you came from afterwards,

220         if(direction[0] == -1){

222         }
223         else if(direction[0] == 1){

225         }
226         else if(direction[1] == -1){
227             extraPlayerMovement[1] = (int)(getHeight()/(2.0*newRadius+1.0))-1;
228         }
229         else if(direction[1] == 1){
230             extraPlayerMovement[1] = 1;
231         }
232         repaint();
233     }

286     public void getNewDrawingRadius(){
287         
288         //reduced vision during night time
289         double dayLight = WeatherSimulator.getSolarIntensity(locPos[0], locPos[1]);
290         
291         if(currentLevel == 0){
292             //radiusExt is a bigger radius due to height effects
293             int radiusExt = radius;
294             //height is all the same in upper/lower layers, so only useful for main map
295             //calculate average height around player and compare it
296             double height = 0;
297             double number = 0;
298             for(int j=-radius;j<radius;j++){
299                 for(int k=-radius;k<radius;k++){
300                     try{
301                         height+=elevation.get(0)[playerPos[0] + j][playerPos[1] + k];
302                         number++;
303                     }catch(ArrayIndexOutOfBoundsException exc){
304                         continue;
305                     }
306                 }
307             }
308             double averageHeight = height/number;
309             if(elevation.get(0)[playerPos[0]][playerPos[1]] > averageHeight){
310                 radiusExt = (int)(radius*elevation.get(0)[playerPos[0]][playerPos[1]]/averageHeight);
311             }
312             
313             if(dayLight > 0){
314                 drawRadius=(int)((0.5+dayLight/1.3)*radiusExt);
315             }
316             else if(dayLight > -0.3){
317                 drawRadius=(int)(radiusExt/2.5);
318             }
319             else if(RPGMain.speler.hasItem("Torch") || RPGMain.speler.hasItem("Lamp")){
320                 drawRadius=(int)(radiusExt/2.0);
321             }
322             else{
323                 drawRadius = 0;
324             }
325         }
326         else{
327             drawRadius = radius;
328         }
329         
330         String weather = WeatherSimulator.getWeather(locPos[0], locPos[1]);
331         
332         double dummyRadius = drawRadius;
333         
334         if(weather.startsWith("misty")){
335             dummyRadius*=0.5;
336         }
337         else if(weather.startsWith("rain")){
338             dummyRadius*=0.8;
339         }
340         else if(weather.startsWith("snow")){
341             dummyRadius*=1.2;
342         }
343         
344         calculateVisibility((int)dummyRadius);
345         
346         //dummyRadius can be bigger because of reflection of snow
347         drawRadius = (int)Math.max(drawRadius, dummyRadius+1);
348         
349         //logger.debug("Calling getNewDrawRadius. dummyradius: " + dummyRadius + "; drawRadius: " + drawRadius + ". Weather: " + weather + ". Daylight: " + dayLight);
350         
351     }

565         }
566         else{
567             g.fillOval(x/* + w*/ + extraPlayerMovement[0], y/* + h*/ + extraPlayerMovement[1], w, h);

641     public void setPlayerMovement(int[] x){
642         //amount of pixels per square
643         int newRadius = Math.max(drawRadius+1, 4);
644         double widthRatio = (double)this.getWidth()/(2*newRadius+1);
645         double heightRatio = (double)this.getHeight()/(2*newRadius+1);
646         double speed = 1.0;
647         boolean[] addedEvents = new boolean[8];
648         
649         double miliseconds = 7000;
650         
651         addKeyListener(this);
652         requestFocus();
653         
654         if(RPGMain.speler.getMovementMode().equalsIgnoreCase("running")){
655             speed = 2.0;
656         }
657         
658         GameFrameCanvas.textField.setEditable(false);
659         
660         boolean addedNoiseCentre = false;
661         
662         walkEvents = new HashMap<String,WalkEvent>();
663         
664         //moving in the x direction
665         if(x[1] == 0){
666             for(int j=0;j<widthRatio;j++){
667                 //move 1 pixel
668                 extraPlayerMovement[0]+=x[0];
669                 try{
670                     // 2/speed seconds in total
671                     Thread.sleep((int)(miliseconds/widthRatio/speed));
672                     
673                     calculateNoiseLevel(widthRatio, speed, addedNoiseCentre, j, miliseconds);
674                     
675                     int direction = Global.generator.nextInt(8);
676                     //logger.debug("probability: " + 0.5/widthRatio);
677                     if(j < widthRatio/2.0 && j%3 == 0){
678                         if(!addedEvents[direction] && Math.random() < 0.25){
679                             logger.debug("Adding event for " + direction);
680                             addEvent(direction);
681                             addedEvents[direction] = true;
682                         }
683                     }
684                 } catch(InterruptedException e){
685                 }
686                 repaint();
687             }
688         }
689         //moving in the y direction
690         else{
691             for(int j=0;j<heightRatio;j++){
692                 //move 1 pixel
693                 extraPlayerMovement[1]+=x[1];
694                 try{
695                     Thread.sleep((int)(miliseconds/heightRatio/speed));
696                     
697                     calculateNoiseLevel(heightRatio, speed, addedNoiseCentre, j, miliseconds);
698                     
699                     int direction = Global.generator.nextInt(8);
700                     if(!addedEvents[direction] && Math.random() < 0.25/heightRatio){
701                         logger.debug("Adding event for " + direction);
702                         addEvent(direction);
703                         addedEvents[direction] = true;
704                     }
705                 } catch(InterruptedException e){
706                 }
707                 repaint();
708             }
709         }
710         //set textfield editable and focused
711         GameFrameCanvas.textField.setEditable(true);
712         GameFrameCanvas.textField.requestFocus();
713         
714         //slide noiselevel back to low values
715         float noiseLevel = GameFrameCanvas.noiseMeter.getNoiseLevel()-0.15f;
716         for(int j=30;j>0;j--){
717             GameFrameCanvas.noiseMeter.addNoiseLevel(-noiseLevel/30.0f);
718             try {
719                 Thread.sleep((int)(500.0/30.0));
720             } catch (InterruptedException e) {
721                 e.printStackTrace();
722             }
723         }
724         
725         extraPlayerMovement[0] = 0;
726         extraPlayerMovement[1] = 0;
727     }

821     public void calculateNoiseLevel(double ratio, double speed, boolean addedNoiseCentre, int j, double miliseconds){
822         double t = miliseconds/speed/ratio*j;
823         double w = 2.0*Math.PI/(miliseconds/speed/8.0);
824         if(RPGMain.speler.getMovementMode().equalsIgnoreCase("walking")){
825             GameFrameCanvas.noiseMeter.setNoiseLevel((float)Math.cos(w*t)*0.05f + 0.5f);
826         }
827         else if(RPGMain.speler.getMovementMode().equalsIgnoreCase("running")){
828             GameFrameCanvas.noiseMeter.setNoiseLevel((float)Math.cos(w*t)*0.10f + 0.7f);
829         }
830         else{
831             GameFrameCanvas.noiseMeter.setNoiseLevel((float)Math.cos(w*t)*0.05f + 0.1f);
832         }
833         if(j%5 == 0 && !addedNoiseCentre){
834             double detectionProb = Math.pow(GameFrameCanvas.noiseMeter.getNoiseLevel(),2.0)/(ratio/5.0);
835             if(Math.random() < detectionProb){
836                 int[] currentPos = RPGMain.speler.getCurrentPosition();
837                 HostileArea currentHA = (HostileArea)Data.wereld[currentPos[0]][currentPos[1]];
838                 currentHA.setPlayerNoiseCentre(new int[]{playerPos[0],playerPos[1],(int)Math.ceil(5.0*GameFrameCanvas.noiseMeter.getNoiseLevel())});
839                 addedNoiseCentre = true;
840             }
841         }

877     public void calculateVisibility(int radius){
878         
879         //mark all spots previously visited as 1
880         for(int j=0;j<width;j++){
881             for(int k=0;k<height;k++){
882                 if(visible[j][k] == 2){
883                     visible[j][k] = 1;
884                 }
885             }
886         }
887         //mark spots in current line of sight as 2
888         for(int k=-radius;k<=radius;k++){
889             for(int j=-radius;j<=radius;j++){
890                 try{
891                     boolean isVisible = false;
892                     
893                     if(Math.sqrt(j*j+k*k) > radius){
894                         isVisible = false;
895                     }
896                     else{
897                         //determine which squares are in line of sight
898                         //the origin is in the playerPos
899                         //set 1 is in the plane x-y
900                         double rico1;
901                         //set 2 is in the plane (player->point)-z
902                         double rico2 = (double)(elevation.get(currentLevel)[playerPos[0] + j][playerPos[1] + k]-elevation.get(currentLevel)[playerPos[0]][playerPos[1]])/(double)(Math.sqrt(j*j + k*k));
903                         
904                         if(j==0){
905                             for(int l=0;l<=Math.abs(k);l++){
906                                 if(((double)elevation.get(currentLevel)[playerPos[0]][playerPos[1] + l*(int)Math.signum(k)] - (double)(rico2*l) - (double)elevation.get(currentLevel)[playerPos[0]][playerPos[1]]) < 0.3){
907                                     isVisible = true;
908                                 }
909                                 else{
910                                     isVisible = false;
911                                     break;
912                                 }
913                             }
914                             if(k==0){
915                                 isVisible = true;
916                             }
917                         }
918                         else{
919                             if(Math.abs(k) > Math.abs(j)){
920                                 rico1 = (double)j/(double)k;
921                                 for(int l=0;l<=Math.abs(k);l++){
922                                     if(((double)elevation.get(currentLevel)[playerPos[0] + (int)(rico1*(double)(Math.signum(k)*l))][playerPos[1] + (int)Math.signum(k)*l] - ((rico2*Math.sqrt(l*l + (int)(rico1*(double)l*rico1*(double)l)))+(double)elevation.get(currentLevel)[playerPos[0]][playerPos[1]])) < 0.3){
923                                         isVisible = true;
924                                     }
925                                     else{
926                                         isVisible = false;
927                                         break;
928                                     }
929                                 }
930                             }
931                             else{
932                                 rico1 = (double)k/(double)j;
933                                 for(int l=0;l<=Math.abs(j);l++){
934                                     if(((double)elevation.get(currentLevel)[playerPos[0] + (int)Math.signum(j)*l][playerPos[1] + (int)(rico1*(double)(Math.signum(j)*l))] - ((rico2*Math.sqrt(l*l + (int)(rico1*(double)l*rico1*(double)l)))+(double)elevation.get(currentLevel)[playerPos[0]][playerPos[1]])) < 0.3){
935                                         isVisible = true;
936                                     }
937                                     else{
938                                         isVisible = false;
939                                         break;
940                                     }
941                                 }
942                             }
943                         }
944                     }
945                     //if it is in line of sight and visible, mark as 2
946                     if(isVisible){
947                             visible[playerPos[0] + j][playerPos[1] + k] = 2;
948                     }
949                 }catch(ArrayIndexOutOfBoundsException exc){
950                     continue;
951                 }
952             }
953         }
954     }
