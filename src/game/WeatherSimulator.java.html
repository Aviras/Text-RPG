36      public static String getWeather(int x, int y){
37          
38          String weather = "clear";
39          double temperature = Data.wereld[x][y].getTemperature();
40          
41          for(Cloud c:clouds){
42              if(new Point(x,y).distance(c.getPosition()[0],c.getPosition()[1]) <= c.getRadius()){
43                  if(c.getRaining()){
44                      if(temperature < 0){
45                          weather = "snow";
46                      }
47                      else{
48                          weather = "rain";
49                      }
50                  }
51                  else if(c.getHeight() - Data.wereld[x][y].getAltitude() < 1){
52                      weather = "misty";
53                  }
54                  else{
55                      weather = "clouded";
56                  }
57              }
58          }
59          
60          double[] windSpeed = new double[2];
61          for(PressurePoint p:pressurePoints){
62              double distance = new Point(x,y).distance(p.getPosition()[0], p.getPosition()[1]);
63              double value = p.getPressure()/distance;
64              windSpeed[0]+=value*((x-p.getPosition()[0])/distance);
65              windSpeed[1]+=value*((y-p.getPosition()[1])/distance);
66          }
67          double windValue = Math.sqrt(Math.pow(windSpeed[0],2) + Math.pow(windSpeed[1], 2));
68          //TODO check values
69          //System.out.println("Wind: " + windValue);
70          if(windValue > 1){
71              weather+=" storm";
72          }
73          else if(windValue > 0.3){
74              weather+=" windy";
75          }
76          else{
77              weather+=" windstill";
78          }
79          
80          return weather;
81          
82      }

172     public static double calcSolarIntensity(int x, int y){
173 
174         //double k = 2*Math.PI/(50*Data.wereld.length); //world map is 1/50 of a wavelength, approx constant over surface
175         double sigma_2 = (double)(4*Data.wereld[0].length); //determines fall-off of gaussian in y-direction
176         
177         try{
178             // reset the temperature every 5 days at noon should things get out of hand
179             if(DayLightThread.getHour() == 12 && RPGMain.speler.getDaysPlayed()%5 == 0){
180                 // 40 e^{(y-(4L + 2L*sin(seasonOffset + w_2*days)))^2/(sqrt(2)*sigma)^2}-15
181                 if(x == 0 && y == 0){
182                     logger.info("Resetting temperature. Cloud #: " + clouds.size() + "; PP #: " + pressurePoints.size());
183                 }
184                 Data.wereld[x][y].setTemperature(40*Math.exp(-Math.pow((y - (4*Data.wereld[0].length + 2.0*Data.wereld[0].length*Math.sin(seasonOffset + w_2*RPGMain.speler.getDaysPlayed())))/(Math.sqrt(2)*sigma_2),2))-15);
185             }
186         } catch(NullPointerException e){
187         }
188         double A = 1;
189         
190         for(Cloud c: clouds){
191             if(Point.distance(x, y, c.getPosition()[0], c.getPosition()[1]) <= c.getRadius()){
192                 A-=c.getMatterAmount()/(Math.pow(c.getRadius(), 2));
193             }
194         }
195         A = Math.max(0.25, A);
196         
197         double heatTransport = A*Math.sin(w_1*(double)DayLightThread.getHour() - Math.PI/2);
198         
199         return heatTransport;

265     public void calculatePressurePoints(){
266         
267         HashMap<double[],Double> candidatePressurePoints = new HashMap<double[],Double>();
268         
269         int heatEquator = (int)(4*Data.wereld[0].length + 2*Data.wereld[0].length*Math.sin(w_2*RPGMain.speler.getDaysPlayed() + seasonOffset));
270         
271         for(int j=0;j<Data.wereld.length;j+=(int)(Data.wereld.length/4.0)){
272             for(int k=0;k<Data.wereld[0].length;k+=(int)(Data.wereld[0].length/3.0)){
273                 // determine pressure, periodical cosine to simulate hadley cells
274                 double value = Math.cos(10*Math.PI/(2*Data.wereld[0].length/5)*(k-heatEquator));
275                 if(Math.abs(value) > 0.7){
276                     candidatePressurePoints.put(new double[] {j,k},value);
277                 }
278             }
279         }
280         // newly inserted points get checked as well, not efficient and not necessary,
281         // so use cloned arraylist in second for-loop
282         ArrayList<PressurePoint> clonedPressurePoints = cloneArrayList(pressurePoints);
283         for(double[] cPPos: candidatePressurePoints.keySet()){
284             boolean alreadyAdded = false;
285             for(PressurePoint p: clonedPressurePoints){
286                 //see if candidate pressure points are close to already existing ones, if so sum their values
287                 if(Point.distance(p.getPosition()[0],p.getPosition()[1], cPPos[0],cPPos[1]) <= 5){
288                     p.addPressure(candidatePressurePoints.get(cPPos));
289                     alreadyAdded = true;
290                     break;
291                 }
292             }
293             //add candidate pressure point that wasn't close to others
294             if(!alreadyAdded){
295                 pressurePoints.add(new PressurePoint(cPPos,candidatePressurePoints.get(cPPos),new double[] {2*Math.random(),2*Math.random()}));
296             }
297         }
298         ArrayList<PressurePoint> delete = new ArrayList<PressurePoint>();
299         for(PressurePoint p: pressurePoints){
300             if(p.getPressure() < 0.2)
301                 delete.add(p);
302         }
303         for(PressurePoint p: delete){
304             pressurePoints.remove(p);
305         }
306         delete.clear();
307         //System.err.println("Amount of pressurePoints: " + pressurePoints.size());
308         candidatePressurePoints.clear();
309     }

313             p.setPosition(new double[]{p.getPosition()[0] + p.getVelocity()[0],p.getPosition()[1] + p.getVelocity()[1]});
314         }

316         // check pressurePoints overlap
317         for(int i=0;i<2;i++){
318             for(int j=0;j<pressurePoints.size();j++){
319                 for(int k=j+1;k<pressurePoints.size();k++){
320                     if(Point.distance(pressurePoints.get(j).getPosition()[0], pressurePoints.get(j).getPosition()[1], pressurePoints.get(k).getPosition()[0], pressurePoints.get(k).getPosition()[1]) <= 5){
321                         pressurePoints.get(j).addPressure(pressurePoints.get(k).getPressure());
322                         pressurePoints.remove(k);
323                         k--;
324                     }
325                 }
326             }
327         }

371     public void calculateCloudMovement(){
372         // use pressurepoints as a 1/rÂ² force field like electrodynamics to calculate velocity
373         double[] velocity = new double[2];
374         double[] cloudPosition = new double[2];
375         ArrayList<Cloud> delete = new ArrayList<Cloud>();
376         int m = 0;
377         for(Cloud c:clouds){
378             cloudPosition = c.getPosition();
379             for(PressurePoint p: pressurePoints){
380                 double[] pressurePosition = p.getPosition();
381                 double distance = Point.distance(pressurePosition[0], pressurePosition[1], cloudPosition[0], cloudPosition[1]);
382                 double value = p.getPressure();
383                 if(m == 0){
384                     ////System.out.println("pressurePoints pos:" + pressurePosition[0] + "," + pressurePosition[1] + ":" + value);
385                 }
386                 velocity[0] += 1.3*value*(cloudPosition[0]-pressurePosition[0])/Math.pow(distance,2);
387                 velocity[1] -= 1.3*value*(cloudPosition[1]-pressurePosition[1])/Math.pow(distance,2);
388             }
389             m++;
390             // clouds move slower over terrain that is higher, and rise to move over them
391             // check the entire area under the cloud
392             int heightDifference = 0;
393             for(int j=-(int)Math.ceil(c.getRadius());j<(int)Math.ceil(c.getRadius());j++){
394                 for(int k=-(int)Math.ceil(c.getRadius());k<(int)Math.ceil(c.getRadius());k++){
395                     if(Math.sqrt(j*j + k*k) <= c.getRadius()){
396                         try{
397                             int newHeightDifference = Data.wereld[(int)(cloudPosition[0] + Math.ceil(velocity[0])) + j][(int)(cloudPosition[1] + Math.ceil(velocity[1])) + k].getAltitude() - c.getHeight();
398                             if(newHeightDifference > 0 && newHeightDifference > heightDifference){
399                                 heightDifference = newHeightDifference;
400                                 //System.err.println("Velocity diminished for cloud at " + cloudPosition[0] + ", " + cloudPosition[1] + ". Height: " + c.getHeight());
401                             }
402                         }catch(NullPointerException e){
403                         }catch(ArrayIndexOutOfBoundsException e){
404                         }
405                     }
406                 }
407             }
408             if(heightDifference > 0){
409                 velocity[0]/=heightDifference+2;
410                 velocity[1]/=heightDifference+2;
411             }
412             ////System.out.println("Velocity for " + c.getPosition()[0] + "," + c.getPosition()[1] + ": " + velocity[0] + "," + velocity[1]);
413             // within world boundaries
414             if(cloudPosition[0] + velocity[0] < Data.wereld.length && cloudPosition[0] + velocity[0] >= 0
415                     && cloudPosition[1] + velocity[1] < Data.wereld[0].length && cloudPosition[1] + velocity[1] >= 0){
416                 ////System.out.println("Cloud from " + cloudPosition[0] + "," + cloudPosition[1] + " to " + (cloudPosition[0]+velocity[0]) + "," + (cloudPosition[1]+velocity[1]) + "; Radius:" + c.getRadius() + ";Height: " + c.getHeight());
417                 c.addPosition(velocity);
418             }
419             else{
420                 if(Math.random() < 0.25){
421                     c.addPosition(new double[]{-velocity[0],-velocity[1]});
422                 }
423                 else{
424                     delete.add(c);
425                 }
426             }
427         }
428         for(Cloud c:delete){
429             clouds.remove(c);
430             //System.out.println("Cloud out of bounds at " + c.getPosition()[0] + "," + c.getPosition()[1]);
431         }
432         logger.info("Clouds size after movement: " + clouds.size());
433         delete.clear();

437     public void calculateRainFall(){
438         // check rainfall, either due to height difference, or critical value of matter density reached
439         // diminish amount of matter, and diminish radius
440         ArrayList<Cloud> delete = new ArrayList<Cloud>();
441         for(Cloud c: clouds){
442             c.setRaining(false);
443             // amount of matter per unit of surface (determined by its radius)
444             double matterDensity = c.getMatterAmount()/(Math.PI*Math.pow(c.getRadius(), 2));
445             ////System.out.println("matterDensity for " + c.getPosition()[0] +"," + c.getPosition()[1] + ": " + matterDensity);
446             // check to see for height differences within cloud radius
447             for(int j=-(int)Math.ceil(c.getRadius());j<=Math.ceil(c.getRadius());j++){
448                 for(int k=-(int)Math.ceil(c.getRadius());k<=Math.ceil(c.getRadius());k++){
449                     if(Math.sqrt(k*k + j*j) <= c.getRadius()){
450                         try{
451                             int heightDifference = Data.wereld[(int)c.getPosition()[0] + j][(int)c.getPosition()[1] + k].getAltitude() - c.getHeight();
452                             if(heightDifference > 0 && matterDensity > 0.5){
453                                 c.addMatterAmount(-0.1*c.getMatterAmount()*(heightDifference));
454                                 c.setRadius(0.9*c.getRadius());
455                                 c.setHeight(c.getHeight()+1);
456                                 c.setRaining(true);
457                                 //System.out.println("Raining at " + c.getPosition()[0] + "," + c.getPosition()[1] + " because of height diff. Height difference " + heightDifference);
458                             }
459                         }catch(NullPointerException e){
460                             continue;
461                         }catch(ArrayIndexOutOfBoundsException e){
462                             continue;
463                         }
464                     }
465                 }
466             }
467             
468             //check for rainfall due to critical density
469             if(matterDensity > 1 && !c.getRaining()){
470                 c.addMatterAmount(-0.40*c.getMatterAmount());
471                 c.setRadius(0.90*c.getRadius());
472                 c.setRaining(true);
473                 //System.out.println("Raining at " + c.getPosition()[0] + "," + c.getPosition()[1] + " because of crit dens. Height: " + c.getHeight() + ", " + matterDensity);
474             }
475             else if(matterDensity < 0.2){
476                 delete.add(c);
477             }
478         }
479         for(Cloud c: delete){
480             clouds.remove(c);
481         }
482         delete.clear();
483     }
