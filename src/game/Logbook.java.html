67      public static void addContent(String path, Integer phase, String content){
68          if(path == null || path.equalsIgnoreCase("") || content == null || content.equalsIgnoreCase("")){
69              return;
70          }
71          logbookPath = path;
72          String[] partialPaths = path.split("/");
73          LogbookNode currentNode = root;
74          for(int j = 0;j<partialPaths.length;j++){
75              boolean mapExists = false;
76              String s = partialPaths[j];
77              ArrayList<LogbookNode> currentChildren = currentNode.getChildren();
78              for(int i=0;i<currentChildren.size();i++){
79                  if(currentChildren.get(i).getName().equalsIgnoreCase(s)){
80                      mapExists = true;
81                      currentNode = currentChildren.get(i);
82                      break;
83                  }
84              }
85              if(!mapExists){
86                  currentNode = currentNode.addChild(s);
87              }
88              if(j == partialPaths.length-1 && content != null){
89                  //phase = -1 is for shortcut
90                  if(phase != -1){
91                      currentNode.addContent(phase,content);
92                  }
93                  else{
94                      currentNode.addShortcut(content);
95                  }
96              }
97          }
98          panel.createPageNumbers();
99  
100     }

105     public static int getPhase(String path){
106         int phase = -1;
107         if(path == null || path.equalsIgnoreCase("")){
108             return phase;
109         }
110         logbookPath = path;
111         String[] partialPaths = path.split("/");
112         LogbookNode currentNode = root;
113         for(int j = 0;j<partialPaths.length;j++){
114             String s = partialPaths[j];
115             ArrayList<LogbookNode> currentChildren = currentNode.getChildren();
116             for(int i=0;i<currentChildren.size();i++){
117                 if(currentChildren.get(i).getName().equalsIgnoreCase(s)){
118                     currentNode = currentChildren.get(i);
119                     break;
120                 }
121             }
122         }
123         phase = currentNode.getPhase();
124         return phase;

345         public void createPageNumbers(){
346             try{
347                 pageNumbers.clear();
348             } catch(NullPointerException exc){
349                 //first time
350                 pageNumbers = new LinkedHashMap<Integer,String>();
351             }
352             
353             int page = 1;
354             
355             //hashmap for the current path, together with the indices for each to know what child to select next
356             LinkedHashMap<LogbookNode,Integer> posInTree = new LinkedHashMap<LogbookNode,Integer>();
357             posInTree.put(root,0);
358             //depth first search through tree
359             //4 is number of categories normally under root dir
360             //should number change, it will always end because root will then be deleted at the first if, and the loop will end
361             while(posInTree.get(root) < 4){
362                 Iterator<LogbookNode> i = posInTree.keySet().iterator();
363                 LogbookNode currentNode = null;
364                 //currentNode is the last LogbookNode added
365                 while(i.hasNext()){
366                     currentNode = i.next();
367                 }
368                 ArrayList<LogbookNode> children = currentNode.getChildren();
369                 //end of the chain, remove the lowest level, and increase the index of the level above it
370                 //check is redundant if you just deleted a level, since the upper level obviously has children
371                 //assuming checking if the index is equal to 0 takes as much time as checking if it's empty
372                 //if the index is equal to the size, that means you checked all children, prevents indexoutofboundsexc
373                 if(children.isEmpty() || posInTree.get(currentNode) == children.size()){
374                     posInTree.remove(currentNode);
375                     if(posInTree.isEmpty()){
376                         //if root gets removed at the end
377                         break;
378                     }
379                     else{
380                         i = posInTree.keySet().iterator();
381                         LogbookNode dummy = null;
382                         while(i.hasNext()){
383                             dummy = i.next();
384                         }
385                         //increase the index of the level above the one you just removed
386                         posInTree.put(dummy, posInTree.get(dummy)+1);
387                     }
388                 }
389                 else{
390                     currentNode = children.get(posInTree.get(currentNode));
391                     //add the lowest lying node to your current pos
392                     posInTree.put(currentNode, 0);
393                     
394                     String path = "";
395                     for(LogbookNode l:posInTree.keySet()){
396                         path+=l.getName() + "/";
397                     }
398                     //leaves the root/, and cuts the extra / at the end
399                     path=path.substring(5, path.length()-1);
400                     pageNumbers.put(page, path);
401                     ////System.out.println("Set " + path + " to page " + page);
402                     page++;
403                     //all pages show two pages
404                     pageNumbers.put(page, path);
405                     ////System.out.println("Set " + path + " to page " + page);
406                     page++;
407                 }
408             }
409             
410

413         public void changePage(int amount){
414             int currentPageNumber = 1;
415             String currentPath = "";
416             for(JButton j:location){
417                 currentPath+=j.getText() + "/";
418             }
419             //cuts the last /
420             currentPath = currentPath.substring(0,currentPath.length()-1);
421             
422             ////System.out.println("Current location: " + currentPath);
423             
424             for(int i:pageNumbers.keySet()){
425                 if(pageNumbers.get(i).equalsIgnoreCase(currentPath)){
426                     currentPageNumber = i;
427                     break;
428                 }
429             }
430             //above always searches for the lowest pageNumber, eg the one on the left, while if amount > 0, so nextPage, it should take the number to the right
431             if(amount > 0){
432                 currentPageNumber++;
433             }
434             String newPath = pageNumbers.get(currentPageNumber + amount);
435             if(newPath == null){
436                 return;
437             }
438             
439             setContent(newPath);

447             }catch(NullPointerException exc){
448                 //first time a search is done

459                 Iterator<LogbookNode> i = posInTree.keySet().iterator();
460                 LogbookNode currentNode = null;
461                 //currentNode is the last LogbookNode added
462                 while(i.hasNext()){
463                     currentNode = i.next();
464                 }
465                 ArrayList<LogbookNode> children = currentNode.getChildren();
466                 //end of the chain, remove the lowest level, and increase the index of the level above it

468                 //assuming checking if the index is equal to 0 takes as much time as checking if it's empty

471                     posInTree.remove(currentNode);
472                     if(posInTree.isEmpty()){
473                         break;
474                     }
475                     else{
476                         i = posInTree.keySet().iterator();

478                         while(i.hasNext()){
479                             dummy = i.next();

485                 else{
486                     currentNode = children.get(posInTree.get(currentNode));
487                     //add the lowest lying node to your current pos
488                     posInTree.put(currentNode, 0);
489                     if(currentNode.getName().toLowerCase().contains(s)){
490                         String path = "";
491                         for(LogbookNode l:posInTree.keySet()){
492                             path+=l.getName() + "/";
493                         }

498                 }

502                 addContent("Search Results/" + split[split.length-1], p);
503             }
504             if(possPaths.isEmpty()){
505                 addContent("Search Results",1,"No results were found.");

507             }
