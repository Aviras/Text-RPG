46      public static void calculateHope(){
47          for(HopeCenter h: hopeCenters){
48              int[] position = h.getPosition();
49              int phase = h.getPhase();
50              double value = h.getValue();
51              for(int j=-phase;j<=phase;j++){
52                  for(int k=-phase;k<=phase;k++){
53                      // emulate a traveling wave, hope doesn't get added continuously,
54                      // would make things less transparent
55                      if(Math.abs(Math.sqrt(j*j + k*k) - phase) < 0.5){
56                          //TODO arrayindexoutofboundsexception
57                          hopeMap[j+position[0]][k+position[1]]+=value/Math.sqrt(j*j+k*k);
58                          if(h.getPlayerInduced()){
59                              reputationMap[j+position[0]][k+position[1]]+=value/Math.sqrt(j*j+k*k);
60                          }
61                      }
62                  }
63              }
64              h.addPhase(1);
65          }
66      }

123     public void checkTermination(){
124         ArrayList<HopeCenter> delete = new ArrayList<HopeCenter>();
125         for(HopeCenter h: hopeCenters){
126             if(h.getPhase() == h.getRadius()){
127                 delete.add(h);
128             }
129         }
130         for(HopeCenter h: delete){
131             hopeCenters.remove(h);
132         }
133         delete.clear();
