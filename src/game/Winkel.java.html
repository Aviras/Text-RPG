46          for(int j=0;j<items.length;j++){
47              if(type.equalsIgnoreCase("equipment")){

51                  waar.add(Data.potions.get(items[j]));

55              }
56              else if(type.equalsIgnoreCase("consumable")){

273     public void trespass() throws InterruptedException{
274         if(WeatherSimulator.getSolarIntensity(position[0], position[1]) > 0){
275             RPGMain.printText(true,"The sun has risen, meaning the smith has returned.");
276             Global.pauseProg(2000);
277             return;
278         }
279         int locked = Global.generator.nextInt(3);
280         if(locked > 1 && !isLocked){
281             RPGMain.printText(true, "The door is locked.");
282             isLocked = true;
283             Global.pauseProg();
284         }
285         
286         busted = false;
287         if(!isLocked){
288             double time = 15;
289             // mean 0.0, std dev 1.0
290             double law = Global.generator.nextGaussian();
291             time = Math.abs(time*law);
292             Busted trespass = new Busted((int)time);
293             RPGMain.printText(true,"You managed to get inside.");
294             while(true){
295                 if(busted == false){
296                     for(int j=0;j<waar.size();j++){
297                         try{
298                             RPGMain.printText(true,(j+1) +": " + waar.get(j).getName() + " worth " +  waar.get(j).getCost() + " gold pieces.");
299                         }catch(NullPointerException ex){
300                         }
301                     }
302                     RPGMain.printText(false,"(Type \"take <index>\" or \"flee\")\n>");
303                     String command = RPGMain.waitForMessage().toLowerCase();
304                     if(busted == true){
305                         command = "flee";
306                     }
307                     if(command.startsWith("take")){
308                         try{
309                             int index = Integer.parseInt(command.substring(5));
310                             RPGMain.speler.addInventoryItem(waar.get(index-1));
311                             RPGMain.printText(true,"You have succesfully stolen " + waar.get(index-1).getName() + ".");
312                             trespass.stopThread();
313                             Global.pauseProg();
314                             break;
315                         } catch(NumberFormatException e){
316                             RPGMain.printText(true,"Index is not a number.");
317                             continue;
318                         }
319                     }
320                     else if(command.equalsIgnoreCase("flee")){
321                         trespass.stopThread();
322                     }
323                 }
324                 break;
325             }
326         }

328     class Busted extends Thread{
329         int sleep;
330         private volatile boolean stop = false;
331         public Busted(int time){
332             sleep = time*1000;
333             start();
334         }
335         public void run(){
336                 try {
337                     sleep(sleep);
338                 } catch (InterruptedException e) {
339                 }
340                 if(stop == false){ // CHECK IF PLAYER HAS ALREADY ESCAPED
341                     RPGMain.recieveMessage("u");
342                     RPGMain.printText(true,"\nYou are caught trespassing!");
343                     busted = true;
344                     String[] telwoorden = {"st","nd","rd"};
345                     RPGMain.speler.addCaughtTrespassing(1);
346                     int timesCaught = RPGMain.speler.getCaughtTrespassing();
347                     try{
348                         RPGMain.printText(true,"This is the " + timesCaught + telwoorden[timesCaught-1] + " time you got caught.");
349                     } catch(ArrayIndexOutOfBoundsException e){
350                         RPGMain.printText(true,"This is the " + timesCaught + "th time you got caught.");
351                     }
352                     /*
353                      * PAYING OF THE FINE
354                      */
355                     double fine = Math.pow((double)timesCaught, (double)timesCaught) + 3.0;
356                     int aGold = RPGMain.speler.getGoud();
357                     RPGMain.printText(true,"Amount to pay: " + fine + ".");
358                     try {
359                         Global.pauseProg(1500);
360                     } catch (InterruptedException e1) {
361                         e1.printStackTrace();
362                     }
363                     if(fine > aGold){
364                         RPGMain.printText(true,"You have an insufficient amount of money to buy off your sentence.\nYou will thereby be sent to prison, your current gold and items removed until the total of the fine is collected.");
365                     }
366                     /*
367                      * TRY TO RANSOM
368                      * 
369                      */
370                     int amountRansom;
371                     while(true){
372                         try{
373                             amountRansom = Integer.parseInt(JOptionPane.showInputDialog(null, "Try to ransom for how much gold?\nYour gold: " + aGold + ".", "Ransom", JOptionPane.QUESTION_MESSAGE));
374                             break;
375                         } catch(NumberFormatException e){
376                         }
377                     }
378                     if(amountRansom > aGold){
379                         amountRansom = aGold;
380                         RPGMain.printText(true,"Amount adjusted to " + aGold);
381                     }
382                     if(amountRansom > 0){
383                         // ABILITY TO RANSOM: RANDOM + CHARISMA + RELATION AMOUNT OFFERED TO FINE
384                         /*
385                          * Stel charisma 4,fine 10,amountRansom 5
386                          * ransom is dan rand + 2 - 1 = rand + 1, gemiddeld 11, 40% succes
387                          */
388                         int ransom = Global.generator.nextInt(20) + (int)0.4*RPGMain.speler.getCharisma() - (int)(fine/2/amountRansom);
389                         if(ransom > 13){
390                             RPGMain.printText(true,"Succesful. You both agree this never happened.\nWhat charisma and some luck can do.");
391                             RPGMain.printText(true,"You lose " + amountRansom + " gold.");
392                             RPGMain.speler.addGoud(-amountRansom);
393                             RPGMain.speler.addCaughtTrespassing(-1);
394                             return;
395                         }
396                         else{ // RANSOM FAILED
397                             RPGMain.printText(true,"The man looks at the money and shakes his head.\nBetter pay up.");
398                             payFine((int)fine,aGold);
399                         }
400                     }
401                     else{// no ransom money
402                         RPGMain.printText(true,"You choose not to resist, and pay the fine.");
403                         payFine((int)fine,aGold);
404                     }
405                     RPGMain.printText(true,"...");
406                 }
407         }
408         private void payFine(int fine,int aGold){
409             RPGMain.speler.addGoud(-(int)fine);
410             int paid = fine;
411             if(fine > aGold){
412                 paid = aGold;
413                 int j = 0;
414                 while((fine-paid) > 0){// LOSE INV ITEMS UNTIL FINE IS PAYED, OR PLAYER IS BROKE
415                     try{
416                         paid+=RPGMain.speler.getInventoryItem(j).getCost();
417                         RPGMain.printText(true,"Lost " + RPGMain.speler.getInventoryItem(j).getName() + ".");
418                         RPGMain.speler.delInventoryItem(j);
419                     } catch(NullPointerException np){
420                         j++;
421                     }
422                     if(j == RPGMain.speler.getInventorySize()){
423                         break;
424                     }
425                 }
426             }
427             RPGMain.printText(true,"Lost " + paid + " gold in cash and property.");
428         }
