19      private static final long serialVersionUID = 8024272791774030091L;
20      private Image portraitImage;
21      Dimension fullDim,imgDim;
22      private int maxBarLength,barLength;
23      private int hunger,thirst,fitness;
24      private boolean showStats = false;
25      
26      public Portrait(String fileName,Dimension fullDim,Dimension imgDim){
27          try {
28              portraitImage = ImageIO.read(new File(fileName));
29          } catch (IOException e) {
30              e.printStackTrace();
31          }
32          this.fullDim = fullDim;
33          this.imgDim = imgDim;
34          maxBarLength= fullDim.width - imgDim.width -10;
35          barLength = maxBarLength;
36          setPreferredSize(fullDim);
37          setOpaque(false);
38          repaint();
39      }
40      
41      public void setBarLength(float percent){
42          barLength=(int)(percent*maxBarLength);
43          repaint();
44      }
45      public void setShowStats(boolean b){
46          showStats = b;
47      }
48      public void setHunger(int x){
49          hunger = x;
50          repaint();
51      }
52      public void setThirst(int x){
53          thirst = x;
54          repaint();
55      }
56      public void setFitness(int x){
57          fitness = x;
58          repaint();
59      }
60      public void paintComponent(Graphics g){
61          super.paintComponent(g);
62          
63          g.drawImage(portraitImage, 0, 0,imgDim.width,imgDim.height, null);
64          g.setColor(Color.red);//background HP bar
65          g.fillRect(imgDim.width + 10, imgDim.height/5, maxBarLength, 7);
66          g.setColor(Color.green); //green HP bar
67          g.fillRect(imgDim.width + 10, imgDim.height/5, barLength, 7);
68          if(showStats){
69              g.setColor(Color.orange);
70              g.drawString("H:" + hunger + "%", imgDim.width + 10, 5*imgDim.height/8);
71              g.setColor(Color.cyan);
72              g.drawString("T:" + thirst + "%", imgDim.width + 70, 5*imgDim.height/8);
73              g.setColor(Color.pink);
74              g.drawString("F:" + fitness + "%", imgDim.width + 10, 7*imgDim.height/8);
75          }
76      }
77      
78      public void changeImage(String fileName){
79          try {
80              portraitImage = ImageIO.read(new File(fileName));
81          } catch (IOException e) {
82              e.printStackTrace();
83          }
84          repaint();
85      }
