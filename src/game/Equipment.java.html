190         // POISONS ARE AWESOME
191         class Poison{
192             private String type,name,description, effectType;
193             private int effect,charges, currentPhase;
194             
195             public Poison(String type, String name, String description, int effect, String effectType, int charges){
196                 this.type = type;
197                 this.name = name;
198                 this.description = description;
199                 this.effect = effect;
200                 this.effectType = effectType;
201                 this.charges = charges;
202                 currentPhase = 0;
203             }
204             public void activate(Wezen victim){
205                 if(currentPhase < charges){
206                     if(type.equalsIgnoreCase("debuff")){
207                         victim.addBuff(name, effectType, effect, charges, 5000, description);
208                     }
209                     else{
210                         RPGMain.printText(true, description.replace("victimName", victim.getName()));
211                         //TODO more different effects
212                         if(effectType.equalsIgnoreCase("damage")){
213                             victim.addHP(-effect);
214                         }
215                         else if(effectType.equalsIgnoreCase("paralysis")){
216                             victim.dimMovement(effect);
217                         }
218                         else if(effectType.endsWith("reduction")){
219                             String stat = effectType.split(" ",2)[0];
220                             if(stat.equalsIgnoreCase("strength")){
221                                 victim.increaseStat(0, -effect);
222                             }
223                             else if(stat.equalsIgnoreCase("dexterity")){
224                                 victim.increaseStat(1, -effect);
225                             }
226                             else if(stat.equalsIgnoreCase("intellect")){
227                                 victim.increaseStat(2, -effect);
228                             }
229                             else if(stat.equalsIgnoreCase("charisma")){
230                                 victim.increaseStat(3, -effect);
231                             }
232                         }
233                     }
234                     currentPhase++;
235                 }
236                 else{
237                     poison = null;
238                 }
239             }
240             public String getDescription(){
241                 return description;
242             }
243             public int getCurrentPhase(){
244                 return currentPhase;
245             }
246             public int getCharges(){
247                 return charges;
248             }
249             public void setCharges(int i){
250                 charges = i;
251             }
252             public void addCharges(int i){
253                 charges+=1;
254             }
255             public void addToCurrentPhase(int i){
256                 currentPhase+=i;
257             }
258             public String getEffectType(){
259                 return effectType;
260             }
261             public String getName(){
262                 return name;
263             }
