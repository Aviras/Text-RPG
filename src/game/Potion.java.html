66          switch(effect){
67          case -1: RPGMain.printText(true, "You gain " + strength + " HP."); RPGMain.speler.addHP(strength); break;
68          default: RPGMain.printText(true, effects[effect]  + " increased by " + strength + " for " + duration + " minutes.");
69                   RPGMain.speler.increaseStat(effect,strength);
