23      public String getContent(File aFile,String[] elName){
24          SAXBuilder parser = new SAXBuilder();
25          try {
26              Document doc = parser.build(aFile);
27              Element root = doc.getRootElement();
28              Element el = root;
29              //iterate on every part of the given path
30              for(int j=0;j<elName.length;j++){
31                  // elName is of format Name,attribute:value
32                  String s = elName[j];
33                  String[] t = s.split(",");
34                      try{
35                          String[] u = t[1].split(":");
36                          @SuppressWarnings("unchecked")
37                          List<Element> children = el.getChildren(t[0]);
38                          //iterate over children to find the one with the right attribute
39                          for(Element e: children){
40                              if(e.getAttributeValue(u[0]).equalsIgnoreCase(u[1])){
41                                  el = e;
42                                  break;
43                              }
44                          }
45                      }catch(ArrayIndexOutOfBoundsException e){
46                          //no attribute specified, child is simply first child with name s
47                          el = el.getChild(s);
48                      }catch(Exception e){
49                          e.printStackTrace();
50                          break;
51                      }
52              }
53              return el.getTextTrim();
54          } catch (JDOMException e) {
55              e.printStackTrace();
56          } catch (IOException e) {
57              e.printStackTrace();
58          } catch(NullPointerException e){
59              e.printStackTrace();
60          }
61          return null;
62      }

64      public Object parseData(String s){
65          // try to split strings into multiple strings in case of multiple IDs
66          // in case of a single int, ints will contain the one int
67          String[] ints = s.split(";");
68          try{
69              ArrayList<Integer> IDs = new ArrayList<Integer>();
70              for(String str: ints){
71                  IDs.add(Integer.parseInt(str));
72              }
73              if(IDs.size() == 1 && !s.contains(";")) return (int)IDs.get(0);
74              return convertIntegers(IDs);
75          } catch(NumberFormatException e){// thrown when s is a string or double, float..
76              ints = null;
77              try{
78                  return Double.parseDouble(s);
79              } catch(NumberFormatException nfe){
80                  if(s.equalsIgnoreCase("true")){
81                      return true;
82                  }
83                  else if(s.equalsIgnoreCase("false")){
84                      return false;
85                  }
86                  return s;
87              }
88          }
89      }

91      public Object loadData(Class<?> cl, Element el){
92          // creates a class of given type, coming from the XML element el
93          
94          ArrayList<Object> constrParam = new ArrayList<Object>();
95          @SuppressWarnings("unchecked")
96          List<Element> properties = el.getChildren();
97          // id is given as attribute for xquery, add id to constrparam first
98          constrParam.add(Integer.parseInt(el.getAttributeValue("id")));
99          Iterator<Element> i = properties.iterator();
100         while(i.hasNext()){
101             Element child = i.next();
102             // if child has more children, those are dealt with in the constructor, child is passed as a parameter
103             if(!child.getChildren().isEmpty()){
104                 constrParam.add(child);
105             }
106             else{
107                 constrParam.add(parseData(child.getText()));
108             }
109         }
110         Class<?>[] paramClass = new Class[constrParam.size()];//soorten parameters
111         for(int j = 0;j<paramClass.length;j++){
112             paramClass[j] = constrParam.get(j).getClass();
113         }
114         try {
115             Constructor<?> objCtor = cl.getConstructor(paramClass);
116             return objCtor.newInstance(constrParam.toArray());
117         } catch (NoSuchMethodException e) {
118             e.printStackTrace();
119             logger.debug("No constructor found for class " + cl.getName() + " for element " + el.getChildText("name"), e);
120         } catch (IllegalArgumentException e) {
121             e.printStackTrace();
122             logger.debug("Error for class " + cl.getName() + " for element " + el.getChildText("name"), e);
123         } catch (InstantiationException e) {
124             e.printStackTrace();
125             logger.debug("Error for class " + cl.getName() + " for element " + el.getChildText("name"), e);
126         } catch (IllegalAccessException e) {
127             e.printStackTrace();
128             logger.debug("Error for class " + cl.getName() + " for element " + el.getChildText("name"), e);
129         } catch (InvocationTargetException e) {
130             e.printStackTrace();
131             logger.debug("Error for class " + cl.getName() + " for element " + el.getChildText("name"), e);
132         }
133         return null;
134     }

146     /**
147      * Change the contents of text file in its entirety, overwriting any
148      * existing text.
149      *
150      * This style of implementation throws all exceptions to the caller.
151      *
152      * @param aFile is an existing file which can be written to.
153      * @throws IllegalArgumentException if param does not comply.
154      * @throws FileNotFoundException if the file does not exist.
155      * @throws IOException if problem encountered during write.
156      */
157     static public void setContents(File aFile, String aContents,boolean append)
158     throws FileNotFoundException, IOException {
159         if (aFile == null) {
160             throw new IllegalArgumentException("File should not be null.");
161         }
162         if (!aFile.exists()) {
163             throw new FileNotFoundException ("File does not exist: " + aFile);
164         }
165         if (!aFile.isFile()) {
166             throw new IllegalArgumentException("Should not be a directory: " + aFile);
167         }
168         if (!aFile.canWrite()) {
169             throw new IllegalArgumentException("File cannot be written: " + aFile);
170         }
171 
172         //use buffering
173         Writer output = new BufferedWriter(new FileWriter(aFile,append));
174         try {
175             //FileWriter always assumes default encoding is OK!
176             output.write( aContents );
177         }
178         finally {
179             output.close();
180         }
181     }
